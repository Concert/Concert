  7 /**
  8  *  A ConcertBackboneCollection is our extension of Backbone.Collection to include
  9  *  functionality that all collections need.
 10  **/
 11 var ConcertBackboneCollection = Backbone.Collection.extend(
 12 	/**
 13 	 *	@scope	ConcertBackboneCollection.prototype
 14 	 **/
 15 {
 16     
 17     initialize: function(models, options) {
 18         this.relatedModel = options.relatedModel;
 19     }, 
 20     
 21     /**
 22      *  Overriding this method allows for each collection to be aware of the master
 23      *  list of model instances, so no duplicate instances are ever created.  We
 24      *  can also save our o2m relationship if options.save is set.
 25      *
 26      *  @param  {Boolean}    options.save    -  Save relation?
 27      *  @param  {Function}    options.error_callback    -    Error callback
 28      *  @param  {String}    options.error_message    Error message
 29      **/
 30     _add : function(model, options) {
 31         options || (options = {});
 32         
 33         var seenInstances = com.concertsoundorganizer.modelManager.seenInstances[this.model.prototype.name];
 34 
 35         /* If the model hasn't yet been instantiated */
 36         if(!(model instanceof Backbone.Model)) {
 37             /* Check with dataset manager to see if it already exists */
 38             var possibleDuplicate = seenInstances.get(model.id);
 39 
 40             /* If there is a duplicate */
 41             if(possibleDuplicate) {
 42                 /* Send the attributes to the duplicate incase there are new ones */
 43                 possibleDuplicate.set(model);
 44 
 45                 /* Use duplicate moving forward */
 46                 model = possibleDuplicate;
 47             }
 48             /* If there is not a duplicate, create new instance */
 49             else {
 50                 model = new this.model(model);
 51                 
 52                 seenInstances.add(model);
 53             }   
 54         }
 55 
 56         model = Backbone.Collection.prototype._add.call(this, model, options);
 57         
 58         if(options.save) {
 59             /* We are 'creating' our relationship (in our modified REST
 60             implementation) */
 61             var method = 'create';
 62             options = com.concertsoundorganizer.helpers.wrapError(options);
 63             options.error = com.concertsoundorganizer.helpers.backboneWrapError(
 64                 options.error, null, options
 65             );
 66             
 67             /* This will be the related URL.  For example, when adding a tag
 68             to an audio segment's list of tags, the url should be something like
 69             "/api/1/audiosegment/1/tag/2/" */
 70             options.url = this.relatedModel.url()+model.url({noBase:true});
 71             
 72             /* we POST to this URL with no other data */
 73             (this.sync || Backbone.sync)(method, null, options);
 74             
 75         }
 76     },
 77     
 78     /**
 79      *  Override internal remove method to allow to save our o2m relationship.
 80      *
 81      *  @param  {Boolean}    options.save    -  Wether or not to save this relation.
 82      *  @param  {Function}    options.error    -    Error callback.
 83      **/
 84     _remove : function(model, options) {
 85         options || (options = {});
 86 
 87         model = Backbone.Collection.prototype._remove.call(this, model, options);
 88 
 89         if(options.save) {
 90             var method = 'delete';
 91             var wrapErrorHelper = com.concertsoundorganizer.helpers.wrapError;
 92             options = wrapErrorHelper(options);
 93 
 94             options.url = this.relatedModel.url()+model.url({noBase:true});
 95 
 96             options.error = com.concertsoundorganizer.helpers.backboneWrapError(options.error, null, options);
 97 
 98             (this.sync || Backbone.sync)(method, null, options);
 99         }
100     },
101      
102      
103     
104 });
