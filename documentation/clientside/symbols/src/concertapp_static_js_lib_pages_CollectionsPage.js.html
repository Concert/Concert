 55     /**
 56      *  When an administrator denies a join request.
 57      *
 58      *  @param  {Request}    req - The request object that the administrator is
 59      *  changing
 60      **/
 61     denyRequestWithConfirm: function(req) {
 62         /* Show a confirm dialog */
 63         com.concertsoundorganizer.notifier.confirm({
 64             title: "Are you sure?", 
 65             content: "Are you sure you want to deny "+req.get('user').get('username')+" from this collection?", 
 66             confirmCallback: function(req){
 67                 return function() {
 68                     req.deny();
 69                 }
 70             }(req)
 71         });
 72     }, 
 73     /**
 74      *  When an administrator approves a join request.
 75      *
 76      *  @param  {Request}    req    -   The request object to be approved.
 77      **/
 78     approveRequestWithConfirm: function(req) {
 79         /* Show a confirm dialog */
 80         com.concertsoundorganizer.notifier.confirm({
 81             title: "Are you sure?", 
 82             content: "Are you sure you want to allow "+req.get('user').get('username')+" to organize this collection?", 
 83             confirmCallback: function(req) {
 84                 return function() {
 85                     req.approve();
 86                 }
 87             }(req) 
 88         });
 89     },
 90     /**
 91      *  When an administrator wishes to delete a collection.
 92      *
 93      *  @param  {Collection}    col    -    The collection we're deleting
 94      **/
 95     deleteCollectionWithConfirm: function(col) {
 96         /* First confirm with the user that this is what they would like to do */
 97         /* TODO: Remove this text */
 98         com.concertsoundorganizer.notifier.confirm({
 99             title: 'Are you sure?', 
100             content: 'Are you sure you want to delete '+col.get('name')+'<br />All associated audio will be removed from Concert.',
101             confirmCallback: function(col) {
102                 return function() {
103                     col.destroy({
104                         success: function(model, response) {
105                         },
106                         error: function(model, response) {
107                             com.concertsoundorganizer.notifier.alert({
108                                 title: 'Error', 
109                                 content: 'An Error occurred: '+response.responseText
110                             });
111                         }
112                     });
113                 };
114             }(col)
115         });
116     },

144         com.concertsoundorganizer.notifier.confirm({
145             title: 'Are you sure?', 
146             content: 'Are you sure you want to leave '+col.get('name')+'?<br />You will no longer be able to work on this collection\'s assets.',
