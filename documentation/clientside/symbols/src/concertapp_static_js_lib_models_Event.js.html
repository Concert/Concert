  7 /**
  8  *  A base class for an event that occurs in the application.
  9  **/
 10 var Event = ConcertBackboneModel.extend(
 11     /**
 12      *  @scope  Event.prototype
 13      **/
 14 
 15 {
 16     foreignKeyAttributes: function() {
 17         return [
 18             {
 19                 attr: 'user', 
 20                 model: User
 21             },
 22             {
 23                 attr: 'collection', 
 24                 model: Collection 
 25             },
 26             {
 27                 attr: 'audioSegment', 
 28                 model: AudioSegment 
 29             },
 30             {
 31                 attr: 'audioFile', 
 32                 model: AudioFile
 33             },
 34             {
 35                 attr: 'tag', 
 36                 model: Tag 
 37             }
 38         ];
 39     }, 
 40     name: 'event', 
 41     /**
 42      *  Override the set method so we can turn our time attribute into an actual
 43      *  date object.
 44      **/
 45     set: function(attrs, options) {
 46         if(attrs && attrs.time) {
 47             attrs.time = new Date(attrs.time);
 48         }
 49         return ConcertBackboneModel.prototype.set.call(this, attrs, options);
 50     }, 
 51 });
 52 
 53 var EventSet = ConcertBackboneCollection.extend(
 54     /**
 55      *  @scope  EventSet.prototype
 56      **/
 57 {
 58     model: Event,
 59     comparator: function(eventModel) {
 60         return -(eventModel.get('time'));
 61     }, 
 62 });
