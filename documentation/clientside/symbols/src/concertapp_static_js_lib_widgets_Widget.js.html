  7 /**
  8  *  This contains general stuff that needs to take place for any widget on the UI.
  9  *	@class
 10  *  @extends    Backbone.View
 11  *
 12  *  @param  {jQuery tmpl object}    template    Template element for this widget
 13  *  @param  {Panel}                 panel       Panel containing this widget
 14  **/
 15 var Widget = Backbone.View.extend(
 16 	/**
 17 	 *	@scope	Widget.prototype
 18 	 **/
 19 {
 20     /**
 21      *  Create the DOM elements associated with this widget using a template.  
 22      *  The widget is not inserted into the DOM in this class.  That is for whoever
 23      *  is instantiating the widget to take care of.
 24      **/    
 25     initialize: function() {
 26         var params = this.options;
 27         
 28         var template = params.template;
 29         if(typeof(template) == 'undefined' && typeof(params.el) == 'undefined') {
 30             throw new Error('params.template or params.el must be undefined');
 31         }
 32         else if(typeof(template) != 'undefined' && template.length == 0) {
 33             throw new Error('template not found');
 34         }
 35         /**
 36          *  Template for this widget.
 37          **/
 38         this.template = template;
 39 
 40         var panel = params.panel;
 41         if(typeof(panel) == 'undefined') {
 42             throw new Error('params.panel is undefined');
 43         }
 44         /**
 45          *  The panel that this widget belongs to.
 46          **/
 47         this.panel = panel;
 48 
 49         _.bindAll(this, "render");
 50         if(this.model) {
 51             this.model.bind('change', this.render);            
 52         }
 53     },
 54     render: function() {
 55         var template = this.template;
 56         if(template) {
 57             /* render new widget */
 58             var newel = template.tmpl(this.model.toJSON());
 59             if(newel.length != 1) {
 60                 throw new Error('widgets must have a single containing element');
 61             }
 62             var el = this.el;
 63             
 64             /* If this element is currently in the DOM */
 65             if($(el).parent().length) {
 66                 /* replace old widget in DOM (or non-dom) */
 67                 $(el).replaceWith(newel);                
 68             }
 69             /* Save reference to new widget */
 70             this.el = newel.get(0);
 71         }
 72         
 73         this.delegateEvents();
 74                 
 75         return this;
 76     }
 77 });
