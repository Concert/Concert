  7 /**
  8  *  Model manager for the organize page.
  9  *  @class
 10  *  @extends    LoggedInModelManager
 11  **/
 12 function OrganizePageModelManager(params) {
 13     if(params) {
 14         this.init(params);
 15     }
 16 }
 17 OrganizePageModelManager.prototype = new LoggedInModelManager();
 18 
 19 /**
 20  *  @constructor
 21  **/
 22 OrganizePageModelManager.prototype.init = function(params) {
 23     LoggedInModelManager.prototype.init.call(this, params);
 24 
 25     var dataToLoad = this._dataToLoad;
 26     
 27     /**
 28      *  The raw audio file data
 29      **/
 30     var fileData = params.files;
 31     if(typeof(fileData) == 'undefined') {
 32         throw new Error('params.files is undefined');
 33     }
 34     dataToLoad.fileData = fileData;
 35     
 36     /* Here we will hold the audio files for this collection (just in case others
 37     are seen) */
 38     this.collectionAudioFiles = new AudioFileSet;
 39     
 40     /**
 41      *  The raw audio segment data
 42      **/
 43     var segmentData = params.segments;
 44     if(typeof(segmentData) == 'undefined') {
 45         throw new Error('params.segments is undefined');
 46     }
 47     dataToLoad.segmentData = segmentData;
 48     
 49     /**
 50      *  Raw tag data
 51      **/
 52     var tagData = params.tags;
 53     if(typeof(tagData) == 'undefined') {
 54         throw new Error('params.tags is undefined');
 55     }
 56     dataToLoad.tagData = tagData;
 57     
 58     /* Here we will hold all of the audio segments for this collection for the
 59     same reason as above */
 60     this.collectionAudioSegments = new AudioSegmentSet;
 61     
 62     /* Here are all of the tags for this collection */
 63     this.collectionTags = new TagSet;
 64 
 65     /* Here we will store the audio segments and files that are selected (from the
 66     audio list panel).  Currently only one segment/file can be selected at once, so 
 67     the total cardinality of these sets will be 1. */
 68     this.selectedAudioSegments = new AudioSegmentSet;
 69     this.selectedAudioFiles = new AudioFileSet;
 70     
 71     /**
 72      *  The current collection we are organizing.
 73      **/
 74     this.collection = new Collection;
 75 };
 76 
 77 OrganizePageModelManager.prototype._loadData = function() {
 78     LoggedInModelManager.prototype._loadData.call(this);
 79     
 80     var dataToLoad = this._dataToLoad;
 81     
 82     /* Load current collection */
 83     this.collection.set(dataToLoad.collectionData);
 84     dataToLoad.collectionData = null;
 85     
 86     /* Most stuff is watching both files and widgets, so do this silently */
 87     this.collectionAudioFiles.refresh(dataToLoad.fileData, {silent: true});
 88     dataToLoad.fileData = null;
 89 
 90     var collectionAudioSegments = this.collectionAudioSegments;
 91     collectionAudioSegments.refresh(dataToLoad.segmentData);
 92     dataToLoad.segmentData = null;
 93 
 94     this.collectionTags.refresh(dataToLoad.tagData);
 95     dataToLoad.tagData = null;
 96 };

 98 /**
 99  *  Use this when files are to be selected on the user interface.
100  *  
101  *  @param  {Array}    params.files    -    The selected audio files
102  *  @param  {Array}    params.segments    - The selected audio segments.
103  **/
104 OrganizePageModelManager.prototype.select_audio = function(params) {
105     var files = params.files;
106     if(typeof(files) == 'undefined') {
107         files = [];
108     }
109     
110     var segments = params.segments;
111     if(typeof(segments) == 'undefined') {
112         segments = [];
113     }
114     
115     var selectedAudioSegments = this.selectedAudioSegments;
116     var selectedAudioFiles = this.selectedAudioFiles;
117     
118     /* Deselect everything */
119     selectedAudioSegments.each(function(seg){
120         seg.set({
121             selected: false 
122         });
123     });
124     selectedAudioFiles.each(function(file){
125         file.set({
126             selected: false 
127         });
128     });
129     
130     /* remove previously selected segments and select new ones */
131     selectedAudioSegments.refresh(segments);
132     /* Remove previously selected files and select new ones */
133     selectedAudioFiles.refresh(files);
134     
135     /* Set all instances to selected */
136     selectedAudioSegments.each(function(seg){
137         seg.set({
138             selected: true
139         });
140     });
141     selectedAudioFiles.each(function(file){
142         file.set({
143             selected: true
144         });
145     });
146 
147 };

149 /**
150  *  Create new audio segment object and set it as currently selected.
151  **/
152 OrganizePageModelManager.prototype.create_and_select_new_segment = function(startTime, endTime) {
153     
154     var selectedSegments = this.selectedAudioSegments;
155     var selectedFiles = this.selectedAudioFiles;
156     
157     /* The audio file that will be the parent for our new segment */
158     var audioFile = null;
159     
160     /* If a segment is currently selected */
161     if(selectedSegments.length) {
162         /* Use segment's parent audio file */
163         audioFile = selectedSegments.first().get('audioFile');
164     }
165     /* If a file is currently selected */
166     else if(selectedFiles.length) {
167         /* Use it as the segment's parent */
168         audioFile = selectedFiles.first();
169     }
170     
171     var timestamp = new Date();
172     
173     /* Create new segment */
174     var newSegment = new AudioSegment({
175         audioFile: audioFile, 
176         beginning: startTime, 
177         end: endTime,
178         /* Creator of the segment is the current user */
179         creator: this.user, 
180         /* For now, name is just timestamp */
181         name: 'segment_'+timestamp.format('yyyy-mm-dd_hh:MM:ss:L'), 
182         /* Collection is current collection */
183         collection: this.collection, 
184     });
185     
186     /* Add to this collection's audio segments */
187     this.collectionAudioSegments.add(newSegment);
188     
189     /* Select new segment */
190     this.page.select_audio({
191         segments: [newSegment], 
192     });
193     
194     /* Now save to server */
195     newSegment.save(null, {
196         /* if there is an error */
197         error_callback: function(newSegment) {
198             return function() {
199                 /* Delete our new segment */
200                 newSegment.destroy();
201             }
202         }(newSegment), 
203         error_message: 'Audio segment was not created.' 
204     });
205 };

207 /**
208  *  When the start and end points of the currently selected segment are changed.
209  *
210  *  @param  {Number}    startTime    The new start time
211  *  @param  {Number}    endTime    The new end time
212  **/
213 OrganizePageModelManager.prototype.modify_current_segment_times = function(startTime, endTime) {
214     /* get current segment */
215     var currentSegment = this.selectedAudioSegments.first();
216     
217     /* Save current values */
218     var oldStartTime = currentSegment.get('beginning');
219     var oldEndTime = currentSegment.get('end');
220     
221     /* update values */
222     currentSegment.set({
223         beginning: startTime, 
224         end: endTime
225     });
226     
227     /* Tell page we've updated */
228     this.page.select_audio({
229         segments: [currentSegment]
230     });
231     
232     /* Try to save */
233     currentSegment.save(null, {
234         /* if stuff fails */
235         error_callback: function(oldStartTime, oldEndTime, currentSegment) {
236             return function() {
237                 /* go back to old times */
238                 currentSegment.set({
239                     beginning: startTime, 
240                     end: endTime 
241                 });
242             };
243         }(oldStartTime, oldEndTime, currentSegment), 
244         error_message: 'Audio segment was not modified' 
245     });
246 };

248 /**
249  *  When an audio segment is to be deleted.
250  *
251  *  @param  {AudioSegment}  segment     The audio segment we're deleting
252  **/
253 OrganizePageModelManager.prototype.delete_audio_segment = function(segment) {
254     var collectionAudioSegments = this.collectionAudioSegments;
255     
256     /* Remove from current collection's audio segments */
257     collectionAudioSegments.remove(segment);
258     
259     /* If this is the currently selected audio segment */
260     if(this.selectedAudioSegments.first() == segment) {
261         /* Select parent audio file */
262         this.page.select_audio({
263             files: [segment.get('audioFile')]
264         });
265     }
266     
267     /* Delete audio segment */
268     segment.destroy({
269         /* If there was a problem */
270         error_callback: function(segment, collectionAudioSegments) {
271             return function() {
272                 /* Put back in audio segments list for current collection */
273                 collectionAudioSegments.add(segment);
274             }
275         }(segment, collectionAudioSegments), 
276         error_message: 'Audio segment was not deleted'
277     });
278 };

280 /**
281  *  When the current segment is to be tagged.
282  *
283  *  @param  {String}    tagName    The name of the tag to give this segment.
284  **/
285 OrganizePageModelManager.prototype.tag_current_segment = function(tagName) {
286     
287     /* Find given tag */
288     var tag = this.collectionTags.find(function(t) {
289         return (t.get('name') == tagName);
290     });
291     
292     var currentSegment = this.selectedAudioSegments.first();
293     
294     /* If tag does not yet exist */
295     if(!tag) {
296         /* Create it */
297         tag = new Tag({
298             /* with the given name */
299             name: tagName, 
300             /* for the current collection */
301             collection: this.collection, 
302             /* Created by our user */
303             creator: this.user, 
304         });
305         
306         /* Save tag */
307         tag.save(null, {
308             /* if there was an error */
309             error_callback: function(model) {
310                 /* delete tag */
311                 model.destroy();
312             }, 
313             error_message: 'Tag was not created.', 
314             success: function(segment, page) {
315                 return function(tag) {
316                     tag.get('segments').add(segment);
317                     segment.get('tags').add(tag, {
318                         /* save changes to server */
319                         save: true, 
320                         /* if error */
321                         error_callback: function(addedTag) {
322                             return function(seg) {
323                                 /* remove tag from segment */
324                                 seg.get('tags').remove(addedTag);
325                                 addedTags.get('segments').remove(seg);
326                             };
327                         }(tag), 
328                         error_message: 'Segment was not tagged.',
329                         success: function(me){
330                             return function( ){
331                                 /* Tell page to re-render for our audio again */
332                                 me.page.select_audio({
333                                     segments: [me.selectedAudioSegments.first()], 
334                                 });
335                             }
336                         }(this), 
337                     });
338                 };
339             }(currentSegment, this.page), 
340         });
341     }
342     else {
343         /* tell tag and segment about eachother */
344         tag.get('segments').add(currentSegment);
345         currentSegment.get('tags').add(tag, {
346             /* save changes to server */
347             save: true, 
348             /* if error */
349             error_callback: function(addedTag) {
350                 return function(seg) {
351                     /* remove tag from segment */
352                     seg.get('tags').remove(addedTag);
353                     addedTags.get('segments').remove(seg);
354                 };
355             }(tag), 
356             error_message: 'Segment was not tagged.'
357         });
358         /* Tell page to re-render for our audio again */
359         this.page.select_audio({
360             segments: [this.selectedAudioSegments.first()], 
361         });
362     }
363     
364     
365 };
