  7 /**
  8  *  Model manager for the organize page.
  9  *  @class
 10  *  @extends    LoggedInModelManager
 11  **/
 12 function OrganizePageModelManager(params) {
 13     if(params) {
 14         this.init(params);
 15     }
 16 }
 17 OrganizePageModelManager.prototype = new LoggedInModelManager();
 18 
 19 /**
 20  *  @constructor
 21  **/
 22 OrganizePageModelManager.prototype.init = function(params) {
 23     LoggedInModelManager.prototype.init.call(this, params);
 24 
 25     var dataToLoad = this._dataToLoad;
 26     
 27     /**
 28      *  Raw collection data for the current collection
 29      **/
 30     var collectionData = params.collectionData;
 31     if(typeof(collectionData) == 'undefined') {
 32         throw new Error('params.collectionData is undefined');
 33     }
 34     dataToLoad.collectionData = collectionData;
 35     
 36     /**
 37      *  The raw audio file data
 38      *
 39     var fileData = params.files;
 40     if(typeof(fileData) == 'undefined') {
 41         throw new Error('params.files is undefined');
 42     }
 43     dataToLoad.fileData = fileData;
 44     
 45     /* Here we will hold the audio files for this collection (just in case others
 46     are seen) 
 47     this.collectionAudioFiles = new AudioFileSet;
 48     
 49     /**
 50      *  The raw audio segment data
 51      *
 52     var segmentData = params.segments;
 53     if(typeof(segmentData) == 'undefined') {
 54         throw new Error('params.segments is undefined');
 55     }
 56     dataToLoad.segmentData = segmentData;
 57     */
 58     /**
 59      *  Raw tag data
 60      *
 61     var tagData = params.tags;
 62     if(typeof(tagData) == 'undefined') {
 63         throw new Error('params.tags is undefined');
 64     }
 65     dataToLoad.tagData = tagData;*/
 66     
 67     /**
 68      *  Raw data for each type of event
 69      **/
 70     var eventDataSetNames = [
 71         'audioSegmentCreatedEventData', 
 72         'audioSegmentTaggedEventData', 
 73         'audioFileUploadedEventData', 
 74         'joinCollectionEventData', 
 75         'leaveCollectionEventData', 
 76         'createCollectionEventData', 
 77         'requestJoinCollectionEventData', 
 78         'requestDeniedEventData', 
 79         'requestRevokedEventData', 
 80         'requestJoinCollectionEventData'
 81     ];
 82     this.eventDataSetNames = eventDataSetNames;
 83     
 84     /* And corresponding sets to hold these events in by type */
 85     var collectionEventSets = [
 86         new AudioSegmentCreatedEventSet,
 87         new AudioSegmentTaggedEventSet,
 88         new AudioFileUploadedEventSet,
 89         new JoinCollectionEventSet,
 90         new LeaveCollectionEventSet,
 91         new CreateCollectionEventSet,
 92         new RequestJoinCollectionEventSet,
 93         new RequestDeniedEventSet,
 94         new RequestRevokedEventSet,
 95         new RequestJoinCollectionEventSet
 96     ];
 97     this.collectionEventSets = collectionEventSets;
 98     
 99 
100     for(var i = 0, il = eventDataSetNames.length; i < il; i++) {
101         var eventDataSetName = eventDataSetNames[i];
102         
103         var eventDataSet = params[eventDataSetName];
104         if(typeof(eventDataSet) == 'undefined') {
105             throw new Error('params['+eventDataSetName+'] is undefined');
106         }
107         dataToLoad[eventDataSetName] = eventDataSet;
108     }
109     
110     
111     /* Here we will hold all of the audio segments for this collection for the
112     same reason as above 
113     this.collectionAudioSegments = new AudioSegmentSet;
114     */
115     /* Here are all of the tags for this collection 
116     this.collectionTags = new TagSet;*/
117 
118     /* Here we will store the audio segments and files that are selected (from the
119     audio list panel).  Currently only one segment/file can be selected at once, so 
120     the total cardinality of these sets will be 1. */
121     this.selectedAudioSegments = new AudioSegmentSet;
122     this.selectedAudioFiles = new AudioFileSet;
123     
124     /**
125      *  The current collection we are organizing.
126      **/
127     this.collection = null;
128 };
129 
130 OrganizePageModelManager.prototype._loadData = function() {
131     LoggedInModelManager.prototype._loadData.call(this);
132     
133     var dataToLoad = this._dataToLoad;
134     
135     /* Load current collection */
136     var collection = this.seenInstances['collection'].get(dataToLoad.collectionData.id);
137     if(!collection) {
138         collection = new Collection();
139     }
140     collection.set(dataToLoad.collectionData);
141     this.collection = collection;
142     dataToLoad.collectionData = null;
143     
144     
145     /* Load data for all events */
146     var collectionEventSets = this.collectionEventSets;
147     var eventDataSetNames = this.eventDataSetNames;
148     
149     for(var i = 0, il = eventDataSetNames.length; i < il; i++) {
150         var eventDataSetName = eventDataSetNames[i];
151         collectionEventSets[i].refresh(dataToLoad[eventDataSetName]);
152         dataToLoad[eventDataSetName] = null;
153     }
154     
155 }

157 /**
158  *  When multiple audio files/segments are to be selected.  Just an interface
159  *  for now.  Method doesn't actually handle multiple selection.
160  *  
161  *  @param  {Array}    params.files    -    The selected audio files
162  *  @param  {Array}    params.segments    - The selected audio segments.
163  **/
164 OrganizePageModelManager.prototype.select_audio = function(params) {
165     var files = params.files;
166     if(typeof(files) == 'undefined') {
167         files = [];
168     }
169     
170     var segments = params.segments;
171     if(typeof(segments) == 'undefined') {
172         segments = [];
173     }
174     
175     /* If there is one audio segment */
176     if(segments.length == 1 && files.length == 0) {
177         this.select_segment(segments[0]);
178     }
179     /* If there is one audio file */
180     else if(files.length == 1 && segments.length == 0) {
181         this.select_file(files[0]);
182     }
183     else {
184         throw new Error('Multiple selection not supported currently.');
185     }
186 
187 };
188 
189 /**
190  *  Called before anything is to be selected.  Handles deselecting of everything.
191  **/
192 OrganizePageModelManager.prototype._deselect_all = function() {
193     var selectedAudioSegments = this.selectedAudioSegments;
194     var selectedAudioFiles = this.selectedAudioFiles;
195     
196     /* Deselect everything */
197     selectedAudioSegments.each(function(seg){
198         seg.set({
199             selected: false 
200         });
201     });
202     selectedAudioFiles.each(function(file){
203         file.set({
204             selected: false 
205         });
206     });
207     
208 };

212 /**
213  *  When a single audio segment is selected.
214  *
215  *  @param  {Number | AudioSegment}    segment - The segment object or id
216  *  @throws selected_segment
217  **/
218 OrganizePageModelManager.prototype.select_segment = function(segment) {
219     /* Clear everything currently selected */
220     this._deselect_all();
221 
222     /* If we were passed an id as a number */
223     if(typeof(segment) == 'number') {
224         /* Get audio segment */
225         segment = this.seenInstances['audiosegment'].get(segment);
226     }
227     
228     /* Set as selected */
229     segment.set({
230         selected: true
231     });
232     
233     
234     /* remove previously selected segments and select new one */
235     this.selectedAudioSegments.refresh([segment]);
236     
237     /* Throw "selected_segment" event */
238     $(this).trigger('select_segment', segment);
239     
240 };
241 
242 /**
243  *  When a single audio file is to be selected.
244  *
245  *  @param  {Number | AudioFile}    file    -    The file or id.
246  *  @throws select_file
247  **/
248 OrganizePageModelManager.prototype.select_file = function(file) {
249     /* Clear everything currently selected */
250     this._deselect_all();
251 
252     /* if we were just passed an id */
253     if(typeof(file) == 'number') {
254         /* First retrieve file instance */
255         file = this.seenInstances['audiofile'].get(file);
256     }
257 
258     file.set({
259         selected: true
260     });
261     
262     
263     /* Remove previously selected files and select new one */
264     this.selectedAudioFiles.refresh([file]);
265     
266     $(this).trigger('select_file', file);
267 };

273 /**
274  *  Create new audio segment object and set it as currently selected.
275  **/
276 OrganizePageModelManager.prototype.create_and_select_new_segment = function(startTime, endTime) {
277     
278     var selectedSegments = this.selectedAudioSegments;
279     var selectedFiles = this.selectedAudioFiles;
280     
281     /* The audio file that will be the parent for our new segment */
282     var audioFile = null;
283     
284     /* If a segment is currently selected */
285     if(selectedSegments.length) {
286         /* Use segment's parent audio file */
287         audioFile = selectedSegments.first().get('audioFile');
288     }
289     /* If a file is currently selected */
290     else if(selectedFiles.length) {
291         /* Use it as the segment's parent */
292         audioFile = selectedFiles.first();
293     }
294     
295     var timestamp = new Date();
296     
297     /* Create new segment */
298     var newSegment = new AudioSegment({
299         audioFile: audioFile, 
300         beginning: startTime, 
301         end: endTime,
302         /* Creator of the segment is the current user */
303         creator: this.user, 
304         /* For now, name is just timestamp */
305         name: 'segment_'+timestamp.format('yyyy-mm-dd_hh:MM:ss:L'), 
306         /* Collection is current collection */
307         collection: this.collection, 
308     });
309     
310     /* Add to this collection's audio segments */
311     this.collectionAudioSegments.add(newSegment);
312     this.seenInstances['audiosegment'].add(newSegment);
313     
314     /* Select new segment */
315     this.page.select_audio({
316         segments: [newSegment], 
317     });
318     
319     /* Now save to server */
320     newSegment.save(null, {
321         /* if there is an error */
322         error_callback: function(newSegment) {
323             return function() {
324                 /* Delete our new segment */
325                 newSegment.destroy();
326             }
327         }(newSegment), 
328         error_message: 'Audio segment was not created.' 
329     });
330 };

332 /**
333  *  When the start and end points of the currently selected segment are changed.
334  *
335  *  @param  {Number}    startTime    The new start time
336  *  @param  {Number}    endTime    The new end time
337  **/
338 OrganizePageModelManager.prototype.modify_current_segment_times = function(startTime, endTime) {
339     /* get current segment */
340     var currentSegment = this.selectedAudioSegments.first();
341     
342     /* Save current values */
343     var oldStartTime = currentSegment.get('beginning');
344     var oldEndTime = currentSegment.get('end');
345     
346     /* update values */
347     currentSegment.set({
348         beginning: startTime, 
349         end: endTime
350     });
351     
352     /* Tell page we've updated */
353     this.page.select_audio({
354         segments: [currentSegment]
355     });
356     
357     /* Try to save */
358     currentSegment.save(null, {
359         /* if stuff fails */
360         error_callback: function(oldStartTime, oldEndTime, currentSegment) {
361             return function() {
362                 /* go back to old times */
363                 currentSegment.set({
364                     beginning: startTime, 
365                     end: endTime 
366                 });
367             };
368         }(oldStartTime, oldEndTime, currentSegment), 
369         error_message: 'Audio segment was not modified' 
370     });
371 };

373 /**
374  *  When an audio segment is to be deleted.
375  *
376  *  @param  {AudioSegment}  segment     The audio segment we're deleting
377  **/
378 OrganizePageModelManager.prototype.delete_audio_segment = function(segment) {
379     var collectionAudioSegments = this.collectionAudioSegments;
380     
381     /* Remove from current collection's audio segments */
382     collectionAudioSegments.remove(segment);
383     
384     /* If this is the currently selected audio segment */
385     if(this.selectedAudioSegments.first() == segment) {
386         /* Select parent audio file */
387         this.page.select_audio({
388             files: [segment.get('audioFile')]
389         });
390     }
391     
392     /* Delete audio segment */
393     segment.destroy({
394         /* If there was a problem */
395         error_callback: function(segment, collectionAudioSegments) {
396             return function() {
397                 /* Put back in audio segments list for current collection */
398                 collectionAudioSegments.add(segment);
399             }
400         }(segment, collectionAudioSegments), 
401         error_message: 'Audio segment was not deleted'
402     });
403 };

405 /**
406  *  When the current segment is to be tagged.
407  *
408  *  @param  {String}    tagName    The name of the tag to give this segment.
409  **/
410 OrganizePageModelManager.prototype.tag_current_segment = function(tagName) {
411     
412     /* Find given tag */
413     var tag = this.collectionTags.find(function(t) {
414         return (t.get('name') == tagName);
415     });
416     
417     var currentSegment = this.selectedAudioSegments.first();
418     
419     /* If tag does not yet exist */
420     if(!tag) {
421         /* Create it */
422         tag = new Tag({
423             /* with the given name */
424             name: tagName, 
425             /* for the current collection */
426             collection: this.collection, 
427             /* Created by our user */
428             creator: this.user, 
429         });
430         
431         /* Add to seen instances */
432         this.seenInstances['tag'].add(tag);
433         /* Add to tags for this collection */
434         this.collectionTags.add(tag);
435     }
436 
437     /* tell tag and segment about eachother */
438     tag.get('segments').add(currentSegment);
439     currentSegment.get('tags').add(tag, {
440         /* save changes to server */
441         save: true, 
442         /* if error */
443         error_callback: function(addedTag) {
444             return function(seg) {
445                 /* remove tag from segment */
446                 seg.get('tags').remove(addedTag);
447                 addedTags.get('segments').remove(seg);
448             };
449         }(tag), 
450         error_message: 'Segment was not tagged.'
451     });
452     /* Tell page to re-render for our audio again */
453     this.page.select_audio({
454         segments: [this.selectedAudioSegments.first()], 
455     });
456     
457     
458     
459 };
