  7 /**
  8  *  Model manager for the organize page.
  9  *  @class
 10  *  @extends    LoggedInModelManager
 11  **/
 12 function OrganizePageModelManager(params) {
 13     if(params) {
 14         this.init(params);
 15     }
 16 }
 17 OrganizePageModelManager.prototype = new LoggedInModelManager();
 18 
 19 /**
 20  *  @constructor
 21  **/
 22 OrganizePageModelManager.prototype.init = function(params) {
 23     LoggedInModelManager.prototype.init.call(this, params);
 24 
 25     var dataToLoad = this._dataToLoad;
 26     
 27     /**
 28      *  The raw audio file data
 29      **/
 30     var fileData = params.files;
 31     if(typeof(fileData) == 'undefined') {
 32         throw new Error('params.files is undefined');
 33     }
 34     dataToLoad.fileData = fileData;
 35     
 36     /* Here we will hold the audio files for this collection (just in case others
 37     are seen) */
 38     this.collectionAudioFiles = new AudioFileSet;
 39     
 40     /**
 41      *  The raw audio segment data
 42      **/
 43     var segmentData = params.segments;
 44     if(typeof(segmentData) == 'undefined') {
 45         throw new Error('params.segments is undefined');
 46     }
 47     dataToLoad.segmentData = segmentData;
 48     
 49     /**
 50      *  Raw tag data
 51      **/
 52     var tagData = params.tags;
 53     if(typeof(tagData) == 'undefined') {
 54         throw new Error('params.tags is undefined');
 55     }
 56     dataToLoad.tagData = tagData;
 57     
 58     /* Here we will hold all of the audio segments for this collection for the
 59     same reason as above */
 60     this.collectionAudioSegments = new AudioSegmentSet;
 61     
 62     /* Here are all of the tags for this collection */
 63     this.collectionTags = new TagSet;
 64 
 65     /* Here we will store the audio segments and files that are selected (from the
 66     audio list panel).  Currently only one segment/file can be selected at once, so 
 67     the total cardinality of these sets will be 1. */
 68     this.selectedAudioSegments = new AudioSegmentSet;
 69     this.selectedAudioFiles = new AudioFileSet;
 70     
 71     
 72 };
 73 
 74 OrganizePageModelManager.prototype._loadData = function() {
 75     LoggedInModelManager.prototype._loadData.call(this);
 76     
 77     var dataToLoad = this._dataToLoad;
 78     
 79     /* Most stuff is watching both files and widgets, so do this silently */
 80     this.collectionAudioFiles.refresh(dataToLoad.fileData, {silent: true});
 81     dataToLoad.fileData = null;
 82     
 83     this.collectionAudioSegments.refresh(dataToLoad.segmentData);
 84     dataToLoad.segmentData = null;
 85     
 86     this.collectionTags.refresh(dataToLoad.tagData);
 87     dataToLoad.tagData = null;
 88 };
 89 
 90 /**
 91  *  Use this when files are to be selected on the user interface.
 92  *  
 93  *  @param  {Array}    params.files    -    The selected audio files
 94  *  @param  {Array}    params.segments    - The selected audio segments.
 95  **/
 96 OrganizePageModelManager.prototype.select_audio = function(params) {
 97     var files = params.files;
 98     if(typeof(files) == 'undefined') {
 99         files = [];
100     }
101     
102     var segments = params.segments;
103     if(typeof(segments) == 'undefined') {
104         segments = [];
105     }
106     
107     /* remove previously selected segments and select new ones */
108     this.selectedAudioSegments.refresh(segments);
109     /* Remove previously selected files and select new ones */
110     this.selectedAudioFiles.refresh(files);
111     
112     /**
113      *  TODO: Refactor code to just use these events.
114      **/
115     if(files.length == 1 && segments.length == 0) {
116         $(this).trigger('audio_file_selected', files[0]);
117     }
118     else if(files.length == 0 && segments.length == 1) {
119         $(this).trigger('audio_segment_selected', segments[0]);
120     }
121     else if(files.length == 0 && segments.length == 0){
122         $(this).trigger('audio_deselected');
123     }
124     else {
125         throw new Error('Not yet implemented multiple selection.');
126     }
127 };
