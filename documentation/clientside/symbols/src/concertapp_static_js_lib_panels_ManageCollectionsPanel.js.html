  7 /**
  8  *  A class for managing the user's collections on the settings page.  
  9  *  @class
 10  *  @extends Panel
 11  **/
 12  
 13  var ManageCollectionsPanel = Panel.extend(
 14 	/**
 15 	 *	@scope	ManageCollectionsPanel.prototype
 16 	 **/
 17 {
 18      initialize: function() {
 19          Panel.prototype.initialize.call(this);
 20 
 21          var params = this.options;
 22 
 23          var contents = this.contents;
 24 
 25          var $ = jQuery;
 26          
 27          /**
 28           *  The panel contents
 29           **/
 30          var manageCollectionsPanelContents = $('#manage_collections_panel_contents');
 31          if(typeof(manageCollectionsPanelContents) == 'undefined') {
 32              throw new Error('manageCollectionsPanelContents is undefined');
 33          }
 34          else if(manageCollectionsPanelContents.length == 0) {
 35              throw new Error('manage_collections_panel_contents not found');
 36          }
 37          this.manageCollectionsPanelContents = manageCollectionsPanelContents;
 38          
 39          /**
 40          *  The set of member collections we are managing on this panel.
 41          **/
 42          var userMemberCollectionsSet = params.userMemberCollectionsSet;
 43          if(typeof(userMemberCollectionsSet) == 'undefined') {
 44              throw new Error('params.userMemberCollectionsSet is undefined');
 45          }
 46          this.userMemberCollectionsSet = userMemberCollectionsSet;
 47          
 48          /**
 49           *  The set of requests we are managing on this panel.
 50           **/
 51          var userRequestsSet = params.userRequestsSet;
 52          if(typeof(userRequestsSet) == 'undefined') {
 53              throw new Error('params.userRequestsSet is undefined');
 54          }
 55          this.userRequestsSet = userRequestsSet;
 56 
 57          _.bindAll(this, "render");
 58          userMemberCollectionsSet.bind('refresh', this.render);
 59          userMemberCollectionsSet.bind('add', this.render);
 60          userMemberCollectionsSet.bind('remove', this.render);
 61          userRequestsSet.bind('refresh', this.render);
 62          userRequestsSet.bind('add', this.render);
 63          userRequestsSet.bind('remove', this.render);
 64      },
 65      render: function() {
 66 
 67          /* Temporary fragment for all DOM additions */
 68          var frag = document.createDocumentFragment();
 69 
 70          this.userMemberCollectionsSet.each(function(panel, frag) {
 71             return function(obj){
 72                 if(obj.get("user_is_admin")) {     
 73                     /* Create a ManageAdminCollectionWidget */
 74                     var widget = new ManageAdminCollectionWidget({
 75                         template: $('#manage_admin_collection_template'), 
 76                         model: obj, 
 77                         panel: panel
 78                     });
 79                 }
 80                 else {
 81                     /* Create a ManageMemberCollectionWidget */
 82                     var widget = new ManageMemberCollectionWidget({
 83                         template: $('#manage_member_collection_template'),
 84                         model: obj,
 85                         panel: panel
 86                     });
 87                 }
 88                 
 89                 frag.appendChild(widget.el);
 90             };
 91         }(this, frag));
 92         
 93         this.userRequestsSet.each(function(panel, frag) {
 94             return function(obj){
 95                 /* Create a ManageRequestWidget */
 96                 var widget = new ManageRequestWidget({
 97                     template: $('#manage_request_collection_template'),
 98                     model: obj,
 99                     panel: panel
100                 });
101                 
102                 frag.appendChild(widget.el);
103             }
104         }(this, frag));
105 
106         /* Update manage collections list */
107         this.manageCollectionsPanelContents.empty().append(frag);
108 
109         return this;
110     }
111 
112 });
