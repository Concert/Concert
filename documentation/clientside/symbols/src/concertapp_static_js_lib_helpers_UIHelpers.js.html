 49 /**
 50  *  Takes an array of DOM nodes, and a container.  It will replace all of the nodes
 51  *  in the container with the given.
 52  *
 53  *  @param  {Array}                 nodes  -   of DOM nodes to add to container
 54  *  @param  {jQuery DOM element}    container
 55  **/
 56 function domElementsReplace(nodes, container) {
 57     /* Empty container */
 58     container.empty();
 59     
 60     /* Loop through all of the nodes, and add them to a fragment */
 61     var frag = document.createDocumentFragment();
 62     for(var i = 0, il = nodes.length; i < il; i++) {
 63         frag.appendChild(nodes[i]);
 64     }
 65     container.append(frag);    
 66 }

 78 /**
 79  *  sec_to_timecode
 80  *  This function will convert between an amount of seconds, and a timecode value.
 81  *
 82  *  @param  {Number}    seconds -   The amount of seconds to convert.
 83  *  @return {String}    hh:mm:ss    -   Formatted timecode string.
 84  **/
 85 function seconds_to_timecode(seconds)
 86 {
 87     if(seconds < 0) {
 88         throw new Error('sec_to_timecode: Error: Seconds cannot be negative.');
 89     }
 90     
 91     var hours = Math.floor(seconds/3600);
 92     var rem = seconds % 3600;
 93     var minutes = Math.floor(rem/60);
 94     var seconds = Math.floor(rem%60);
 95     /* pad zeros */
 96     if(hours < 10)
 97     {
 98         hours = '0'+hours;
 99     }
100     if(minutes < 10)
101     {
102         minutes = '0'+minutes;
103     }
104     if(seconds < 10)
105     {
106         /* pad to beginning */
107         seconds = '0'+seconds;
108     }
109     
110     return hours+':'+minutes+':'+seconds;
111 }

113 /**
114  *  Takes an ISO time and returns a string representing how long ago the date 
115  *  represents.
116  *
117  *  @param  {Date}    date    The Date object we are converting
118  **/
119 function pretty_date(date){
120     /*
121      * JavaScript Pretty Date
122      * Copyright (c) 2008 John Resig (jquery.com)
123      * Licensed under the MIT license.
124      */
125     
126     //var date = new Date((time || "").replace(/-/g,"/").replace(/[TZ]/g," ")),
127     var diff = (((new Date()).getTime() - date.getTime()) / 1000),
128     day_diff = Math.floor(diff / 86400);
129 
130     if ( isNaN(day_diff) || day_diff < 0 || day_diff >= 31 )
131     return;
132 
133     return day_diff == 0 && (
134         /* If less than a minute ago, "just now" */
135         diff < 60 && "just now" ||
136         /* If less than 2 minutes ago, "1 minute ago" */
137         diff < 120 && "1 minute ago" ||
138         /* If less than 1 hour ago, "5 minutes ago" */
139         diff < 3600 && Math.floor( diff / 60 ) + " minutes ago" ||
140         /* If less than 2 hours ago, "1 hour ago" */
141         diff < 7200 && "1 hour ago" ||
142         /* If less than 1 day ago, "2 hours ago" */
143         diff < 86400 && Math.floor( diff / 3600 ) + " hours ago") ||
144         /* If 1 day ago, "Yesterday at 4:25pm" */
145         day_diff == 1 && "Yesterday at "+date.format("h:MMtt") ||
146         /* If less than 5 days ago, "Monday at 4:25pm" */
147         day_diff < 5 && date.format("dddd") + " at " + date.format("h:MMtt") ||
148         /* else, "April 5 at 4:25pm" */
149         date.format("mmmm d") + " at " + date.format("h:MMtt");
150 //        day_diff < 7 && day_diff + " days ago" ||
151 //        day_diff < 31 && Math.ceil( day_diff / 7 ) + " weeks ago";
152 }
