  7 /**
  8  *  The widget that displays the playhead superimposed on the detail waveform.
  9  *  @class
 10  *  @extends    Component
 11  **/
 12 var WaveformPlayheadComponent = Component.extend(
 13 	/**
 14 	 *	@scope	WaveformPlayheadComponent.prototype
 15 	 **/
 16 {
 17     initialize: function() {
 18         Component.prototype.initialize.call(this);
 19 
 20         var params = this.options;
 21         
 22         /* The HTML5 audio element that we will watch */
 23         var audio = params.audio;
 24         if(typeof(audio) == 'undefined') {
 25             throw new Error('params.audio is undefined');
 26         }
 27         this.audio = audio;
 28         
 29         var pxPerSecond = 10;
 30         this.pxPerSecond = pxPerSecond;
 31         
 32         this.fileDuration = null;
 33         
 34         /* As the audio plays, animate playhead */
 35 //        $(audio).bind('timeupdate', function(me) {
 36         setInterval(function(me){
 37             return function() {
 38                 /* TODO: inefficient really, we already have access to currentTime
 39                 here */
 40                 me.animate();
 41             };
 42         }(this), 200);
 43     },
 44 
 45     /**
 46      *  audio_file_selected is called from the panel 
 47      *  when a new audio file is selected
 48      *  obtains duration from the selectedAudioFile
 49      *  @param  {AudioFile}    selectedAudioFile    -   audio file that was selected
 50      **/
 51     audio_file_selected: function(selectedAudioFile) {
 52         var duration = selectedAudioFile.get('duration');
 53         this.fileDuration = duration;
 54     },
 55     
 56     /**
 57      *  called from panel when new audio segment is selected.
 58      *
 59      *  @param  {AudioSegment}    selectedAudioSegment    - audio segment selected
 60      **/
 61     audio_segment_selected: function(selectedAudioSegment) {
 62         this.fileDuration = selectedAudioSegment.get('audioFile').get('duration');
 63     }, 
 64 
 65     /**
 66      *  animate is called from event handler above.
 67      *  moves the playhead to the left the currentTime * 10
 68      **/
 69     animate: function() {
 70         var leftPx = this.audio.currentTime * this.pxPerSecond
 71         this.el.css('left', leftPx);
 72         
 73         return leftPx;
 74     },
 75     
 76     reset: function() {
 77         this.el.css('left', '0px');
 78     },
 79     
 80     position: function() {
 81         return this.el.position().left;
 82     },
 83 });
