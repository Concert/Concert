 10 function CollectionsPageModelManager(params) {
 11     if(params) {
 12         this.init(params);
 13     }
 14 }
 15 CollectionsPageModelManager.prototype = new LoggedInModelManager();
 16 
 17 /**
 18  *  @constructor
 19  **/
 20 CollectionsPageModelManager.prototype.init = function(params) {
 21     LoggedInModelManager.prototype.init.call(this, params);
 22 
 23     var dataToLoad = this._dataToLoad;
 24         
 25     /** The raw collection data for the collections that the current user has
 26         requested to join **/
 27     var requestData = params.requests;
 28     if(typeof(requestData) == 'undefined') {
 29         throw new Error('params.requests is undefined');
 30     }
 31     dataToLoad.requestData = requestData;
 32     
 33     /*  Backbone set that will hold the Concert Collection objects
 34         that the user has requested to join */
 35     this.userRequests = new RequestSet;
 36     
 37     
 38 };
 39 
 40 CollectionsPageModelManager.prototype._loadData = function() {
 41     LoggedInModelManager.prototype._loadData.call(this);
 42     
 43     var dataToLoad = this._dataToLoad;
 44     
 45     var seenInstances = com.concertsoundorganizer.modelManager.seenInstances['collection'];
 46     
 47     var userMemberCollections = this.userMemberCollections;
 48     for(var i=0, il = userMemberCollections.length; i < il; i++) {
 49         if(userMemberCollections.at(i).get('admin') == this.user) {
 50             userMemberCollections.at(i).set({'user_is_admin': true});
 51         }
 52     }
 53             
 54     var seenRequests = this.seenInstances['request'];
 55     this.userRequests.refresh(dataToLoad.requestData);
 56     dataToLoad.requestData = null;
 57 };
 58 
 59 /**
 60  *  Create a new collection.  The user who is logged in is the administrator, as
 61  *  there is no other way to create a collection in Concert.
 62  *
 63  *  @param  {String}    name    -   The name of the new collection.
 64  **/
 65 CollectionsPageModelManager.prototype.create_new_collection = function(name) {
 66     var user = this.user;
 67     
 68     /* Create new collection */
 69     var newCollection = new Collection({
 70         name: name,
 71         users: new UserSet(user),
 72         admin: user,
 73         'user_is_admin': true
 74     });
 75     
 76     /* Add to collections */
 77 //    this.userAdminCollections.add(newCollection); - we no longer hvae userAdminCollections
 78     this.userMemberCollections.add(newCollection);
 79     
 80 
 81     /* Save to server */
 82     newCollection.save(null, {
 83         error_callback: function(newCollection) {
 84             return function() {
 85                 newCollection.destroy();
 86             }
 87         }(newCollection), 
 88         error_message: 'Collection was not created'
 89     });
 90 };
 91 
 92 /**
 93  *  Request to join a collection.  This can be moved to a parent class if the
 94  *  functionality is desired in other pages.
 95  *
 96  *  @param  {Collection}    col    -    The Concert Collection that we are requesting
 97  *  to join.
 98  **/
 99 CollectionsPageModelManager.prototype.request_to_join = function(col) {
100     var newRequest = new Request({
101         collection: col, 
102         user: this.user
103     });
104     
105     newRequest.save(null, {
106         error_callback: function(newRequest) {
107             return function() {
108                 newRequest.destroy();                
109             };
110         }(newRequest),
111         error_message: 'Request was not sent'
112     });
113     
114     this.userRequests.add(newRequest);
115 };
