  7 /**
  8  *  The panel that contains the upload form for the user to upload multiple files.
  9  *  @class
 10  **/
 11 var UploadPanel = Panel.extend(
 12 	/**
 13 	 *	@scope	UploadPanel.prototype
 14 	 **/
 15 {
 16     initialize: function() {
 17         Panel.prototype.initialize.call(this);
 18 
 19         var params = this.options;
 20         
 21         var contents = this.contents;
 22         var $ = jQuery;
 23         
 24         /* We will keep the filewidgets here, indexed by their 'count' */
 25         this.files = [];
 26 
 27         /* We will also keep the filewidgets here, but remove them when they start
 28             uploading */
 29         this.filesToUpload = [];
 30 
 31         /* The form we will change each time we upload a file */
 32         var uploadFormElement = $('#server_upload_form');
 33         if(typeof(uploadFormElement) == 'undefined') {
 34             throw new Error('params.uploadFormElement is undefined');
 35         }
 36         else if(uploadFormElement.length == 0) {
 37             throw new Error('uploadFormElement not found');
 38         }
 39         this.uploadFormElement = uploadFormElement;
 40 
 41         var fileChooserContainer = $('#choose_file_container');
 42         if(typeof(fileChooserContainer) == 'undefined') {
 43             throw new Error('params.fileChooserContainer is undefined');
 44         }
 45         else if(fileChooserContainer.length == 0) {
 46             throw new Error('fileChooserContainer not found');
 47         }
 48         this.fileChooserContainer = fileChooserContainer;
 49 
 50         var fileChooserTemplate = $('#file_chooser_template');
 51         if(typeof(fileChooserTemplate) == 'undefined') {
 52             throw new Error('params.fileChooserTemplate is undefined');
 53         }
 54         else if(fileChooserTemplate.length == 0) {
 55             throw new Error('fileChooserTemplate not found');
 56         }
 57         this.fileChooserTemplate = fileChooserTemplate;
 58 
 59         var uploadFileWidgetTemplate = $('#uploadfilewidget_template');
 60         if(typeof(uploadFileWidgetTemplate) == 'undefined') {
 61             throw new Error('params.uploadFileWidgetTemplate is undefined');
 62         }
 63         else if(uploadFileWidgetTemplate.length == 0) {
 64             throw new Error('uploadFileWidgetTemplate not found');
 65         }
 66         this.uploadFileWidgetTemplate = uploadFileWidgetTemplate;
 67 
 68         var filesTable = $('#files_table');
 69         if(typeof(filesTable) == 'undefined') {
 70             throw new Error('params.filesTable is undefined');
 71         }
 72         else if(filesTable.length == 0) {
 73             throw new Error('filesTable not found');
 74         }
 75         this.filesTable = filesTable;
 76 
 77         var uploadButtonElement = $('#upload_submit_button');
 78         if(typeof(uploadButtonElement) == 'undefined') {
 79             throw new Error('params.uploadButtonElement is undefined');
 80         }
 81         else if(uploadButtonElement.length == 0) {
 82             throw new Error('uploadButtonElement not found');
 83         }
 84         this.uploadButtonElement = uploadButtonElement;
 85 
 86         var fileChooserLabel = $('#file_uploader_label');
 87         if(typeof(fileChooserLabel) == 'undefined') {
 88             throw new Error('params.fileChooserLabel is undefined');
 89         }
 90         else if(fileChooserLabel.length == 0) {
 91             throw new Error('fileChooserLabel not found');
 92         }
 93         this.fileChooserLabel = fileChooserLabel;
 94 
 95         var statusContainer = $('#status_container');
 96         if(typeof(statusContainer) == 'undefined') {
 97             throw new Error('params.statusContainer is undefined');
 98         }
 99         else if(statusContainer.length == 0) {
100             throw new Error('statusContainer not found');
101         }
102         this.statusContainer = statusContainer;
103 
104         /* This will animate the dots on the status container */
105         this.dotsAnimator = new AnimatedDots({
106             container: $('#status_animated_dots')
107         });
108 
109         /* We don't need to show status container right now */
110         statusContainer.hide();
111 
112 
113         /* When upload button is pressed, start uploading */
114         uploadButtonElement.click(function(me) {
115             return function() {
116                 me.startUpload();
117             };
118         }(this));
119 
120 
121         /* The current file chooser that the user can see */
122         this.currentFileChooser = null;
123 
124         /* Keep track of how many files have been chosen.  This is incremented
125             when a file is chosen. */
126         this.fileIndex = -1;
127 
128 
129         this.createNewFileChooser();
130         
131 
132         _.bindAll(this, "render");
133         this.render();
134     },
135     render: function() {
136         
137         return this;
138     },
139     handleUploadError: function() {
140         /* TODO: Write this function */
141     },
142     /**
143      *  Increment this.count member variable, and create the new "file chooser".  Then
144      *  put this in the appropriate location in the DOM.
145      **/
146     createNewFileChooser: function() {
147         /* Increment the file index because we may add a new file */
148         var newFileIndex = ++this.fileIndex;
149 
150         /* Create new file chooser */
151         var chooser = this.fileChooserTemplate.tmpl({
152             id: newFileIndex, 
153         });
154 
155         /* When a file is chosen with this new element */
156         chooser.bind('change', function(me){
157             return function() {
158                 me.addFile($(this));
159             };
160         }(this));
161 
162         /* Add chooser to DOM */
163         this.fileChooserContainer.html(chooser);    
164 
165     },
166     /**
167      *  This should be called when the user selects a file.  It will add the file to 
168      *  the queue, as well as to the 'file_list' in the DOM.  This will be done by
169      *  creating a new UploadingFileWidget, and moving the inputElement into the
170      *  widget.
171      *
172      *  @param  {jQuery HTMLInputElement}   fileChooser - the <input> element that 
173      *                                      the user used to select this file.
174      **/
175     addFile: function(fileChooser) {
176         /* The index for the current file */
177         var index = this.fileIndex;
178 
179         /* Create new UploadFileWidget */
180         var widget = new UploadFileWidget({
181             template: this.uploadFileWidgetTemplate, 
182             model: new Backbone.Model({
183                 filename: fileChooser.attr('files')[0].fileName, 
184                 id: index
185             }),
186             /* Remove fileChooser from DOM, and send to UploadFileWidget */
187             fileChooser: fileChooser.detach(), 
188             panel: this
189         });
190 
191 
192         /* Add widget to DOM */
193         this.filesTable.append($(widget.render().el));
194         /* Add widget to data structures */
195         this.files[index] = widget;
196         this.filesToUpload[index] = widget;
197 
198         /* Re-populate document with new file chooser (do this when we can support
199             multiple file uploads) */
200         //this.createNewFileChooser();
201 
202         /* Remove "select a file to upload" button */
203         this.fileChooserLabel.hide();
204     },
205     /**
206      *  Should be called when upload button is pressed.  This will initiate the uploading
207      *  of all the widgets in the queue.
208      **/
209     startUpload: function() {
210 
211         /* Hide upload button */
212         this.uploadButtonElement.hide();
213 
214         /* Hide file chooser and file chooser label form */
215         this.fileChooserContainer.hide();
216         this.fileChooserLabel.hide();
217 
218         /* Show "uploading..." or something of that nature at the top of the file list */
219         this.statusContainer.show();
220 
221         /* Start dot animation */
222         this.dotsAnimator.start();
223 
224         /* Upload first file */
225         this.uploadNext();
226     },
227     /**
228      *  This should be called after startUpload has been called once, and after
229      *  each file is done uploading.
230      **/
231     uploadNext: function() {
232         var next = this.filesToUpload.shift();
233 
234         next.startUpload();
235     }
236 });
