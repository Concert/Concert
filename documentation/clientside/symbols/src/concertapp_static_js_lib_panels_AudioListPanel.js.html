  7 /**
  8  *  Displays the segments and files within the audio organize page.
  9  *	@class
 10  *  @extends    Panel
 11  **/
 12 var AudioListPanel = Panel.extend(
 13 	/**
 14 	 *	@scope	AudioListPanel.prototype
 15 	 **/
 16 {
 17     initialize: function() {
 18         Panel.prototype.initialize.call(this);
 19         
 20         var params = this.options;
 21         
 22         /**
 23          *  The set of files we're watching for this panel
 24          *
 25         var files = params.files;
 26         if(typeof(files) == 'undefined') {
 27             throw new Error('params.files is undefined');
 28         }
 29         this.files = files;
 30 */
 31         /**
 32          *  The set of segments we're watching for this panel
 33          *
 34         var segments = params.segments;
 35         if(typeof(segments) == 'undefined') {
 36             throw new Error('params.segments is undefined');
 37         }
 38         this.segments = segments;
 39 */
 40         
 41         /**
 42          *  The template for each file in the list
 43          **/
 44         var fileWidgetTemplate = $('#file_widget_template');
 45         if(typeof(fileWidgetTemplate) == 'undefined') {
 46             throw new Error('$(\'#file_widget_template\') is undefined');
 47         }
 48         else if(fileWidgetTemplate.length == 0) {
 49             throw new Error('fileWidgetTemplate not found');
 50         }
 51         this.fileWidgetTemplate = fileWidgetTemplate;
 52         
 53         
 54         /**
 55          *  The template for each segment in the list
 56          **/
 57         var segmentWidgetTemplate = $('#segment_widget_template');
 58         if(typeof(segmentWidgetTemplate) == 'undefined') {
 59             throw new Error('$(\'#segment_widget_template\') is undefined');
 60         }
 61         else if(segmentWidgetTemplate.length == 0) {
 62             throw new Error('segmentWidgetTemplate not found');
 63         }
 64         this.segmentWidgetTemplate = segmentWidgetTemplate;
 65 
 66         
 67         /**
 68          *  If a segment/file has been selected, the model manager will throw
 69          *  events.
 70          **/
 71         var modelManager = params.modelManager;
 72         if(typeof(modelManager) == 'undefined') {
 73             throw new Error('params.modelManager is undefined');
 74         }
 75         /* But we don't need to keep the model manager in memory
 76         this.modelManager = modelManager;*/
 77         
 78         /* The widget that represents the currently selected audio file/segment */
 79         this.selectedWidget = null;
 80         
 81         /* The list of our file widgets, indexed by ID of file object */
 82         this.fileWidgets = {};
 83         
 84         /* The list of segment widgets, indexed by ID of segment object */
 85         this.segmentWidgets = {};
 86         
 87         
 88         _.bindAll(this, "render");
 89         segments.bind('refresh', this.render);
 90         segments.bind('add', this.render);
 91         segments.bind('remove', this.render);
 92         files.bind('refresh', this.render);
 93         files.bind('add', this.render);
 94         files.bind('remove', this.render);
 95     }, 
 96     
 97     /**
 98      *  Called when segments or file list is changed.
 99      **/
100     render: function() {
101         
102         /* temporary frag for dom additions */
103         var frag = document.createDocumentFragment();
104         
105         /* Empty our list of segment and file widgets */
106         var fileWidgets = {};
107         var segmentWidgets = {};
108         
109         /* Put each file in list */
110         this.files.each(function(fileWidgetTemplate, panel, frag, fileWidgets) {
111             return function(obj) {
112                 /* Create a file widget */
113                 var widget = new FileWidget({
114                     template: fileWidgetTemplate, 
115                     model: obj, 
116                     panel: panel,
117                 });
118                 
119                 fileWidgets[obj.get('id')] = widget;
120                 
121                 frag.appendChild(widget.el);
122             };
123         }(this.fileWidgetTemplate, this, frag, fileWidgets));
124         
125         
126         /* Put each segment in list */
127         this.segments.each(function(segmentWidgetTemplate, panel, frag, segmentWidgets) {
128             return function(obj) {
129                 /* Create segment widget */
130                 var widget = new SegmentWidget({
131                     template: segmentWidgetTemplate,
132                     model: obj,
133                     panel: panel, 
134                 });
135                 
136                 segmentWidgets[obj.get('id')] = widget;
137                 
138                 frag.appendChild(widget.el);
139             };
140         }(this.segmentWidgetTemplate, this, frag, segmentWidgets));
141                 
142         /* update panel */
143         this.contents.html(frag);
144         
145         /* Save list of file/segment widgets */
146         this.segmentWidgets = segmentWidgets;
147         this.fileWidgets = fileWidgets;
148     }, 
149 });
