  7 /**

 13 /**
 14  *  This class includes functionality that runs on each page that the user is logged
 15  *  in.
 16  *  @class
 17  *  @extends    Page
 18  **/
 19 var LoggedInPage = Page.extend(
 20 	/**
 21 	 *	@scope	LoggedInPage.prototype
 22 	 **/
 23 {
 24     
 25     /**
 26      *  Our dataset manager is this one.
 27      **/
 28     _initializeModelManager: function(params) {
 29         return new LoggedInModelManager(params);
 30     }, 
 31     
 32     /**
 33      *  Every page where a user is logged in will display a global options panel.
 34      **/
 35     _initializeViews: function() {
 36         Page.prototype._initializeViews.call(this);
 37         
 38         /*  Create the globalOptionsPanel (the buttons and menus at the top of every 
 39             page) */
 40         this.globalOptionsPanel = new GlobalOptionsPanel({
 41             page: this, 
 42             el: $('#global_options_panel'),
 43             userMemberCollections: this.modelManager.userMemberCollections
 44         });
 45     },
 46     _initialize_routes: function() {
 47         Page.prototype._initialize_routes.call(this);
 48         
 49         return;
 50     }, 
 51 });
