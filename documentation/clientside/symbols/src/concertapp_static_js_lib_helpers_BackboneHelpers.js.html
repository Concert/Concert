1 /**
  2 *  Override of Backbone.sync so we can handle related fields better.
  3 **/
  4 
  5 // Map from CRUD to HTTP for our default `Backbone.sync` implementation.
  6 var methodMap = {
  7     'create': 'POST',
  8     'update': 'PUT',
  9     'delete': 'DELETE',
 10     'read'  : 'GET'
 11 };
 12 
 13 // Helper function to get a URL from a Model or Collection as a property
 14 // or as a function.
 15 var getUrl = function(object) {
 16   if (!(object && object.url)) return null;
 17   return _.isFunction(object.url) ? object.url() : object.url;
 18 };
 19 
 20 
 21 Backbone.sync = function(method, model, options) {
 22     var type = methodMap[method];
 23     
 24     // Default JSON-request options.
 25     var params = _.extend({
 26         type:         type,
 27         contentType:  'application/json',
 28         dataType:     'json',
 29         processData:  false
 30     }, options);
 31 
 32     // Ensure that we have a URL.
 33     if (!params.url) {
 34         params.url = getUrl(model) || urlError();
 35     }
 36     
 37 
 38     // Ensure that we have the appropriate request data.
 39     if (!params.data && model && (method == 'create' || method == 'update')) {
 40         var data = model.toJSON();
 41         
 42         for(var key in data) {
 43             var attr = data[key];
 44             /* This makes sure that only the URL of a related object is sent */
 45             if(attr instanceof Backbone.Model) {
 46                 data[key] = data[key].url();
 47             }
 48             /* This makes sure that only URLS of related objects in 
 49             collections are sent */
 50             else if(attr instanceof Backbone.Collection) {
 51                 /* Create list of urls */
 52                 data[key] = [];
 53                 attr.each(function(result){
 54                     return function(obj) {
 55                         result.push(obj.url());
 56                     }
 57                 }(data[key]));
 58             }
 59         }
 60         params.data = JSON.stringify(data);
 61     }
 62 
 63     // Make the request.
 64     $.ajax(params);
 65 };

 67 /**
 68  *  Helps to display error to user.
 69  **/
 70 com.concertsoundorganizer.helpers.wrapError = function(options) {
 71     if(options){
 72         /* Get error messsage provided from caller */
 73         var error_message = options.error_message;        
 74     }
 75     else {
 76         options = {};
 77     }
 78 
 79     if(typeof(error_message) == 'undefined') {
 80         var error_message = 'An error has occurred';
 81     }
 82 
 83     /* Wrap error callback */
 84     options.error = function(error_message, callback) {
 85         return function(model, resp) {
 86             var responseText = resp.responseText;
 87             if(resp && responseText != '') {
 88                 resp = JSON.parse(responseText);                 
 89             }
 90             else {
 91                 resp = {};
 92             }
 93 
 94             if(resp.error_message) {
 95                 error_message += ': '+resp.error_message;
 96             }
 97             else {
 98                 error_message += '.';
 99             }
100 
101             /* display error to the user */
102             com.concertsoundorganizer.notifier.alert({
103                 title: 'Error', 
104                 content: error_message
105             });
106 
107             if(callback) {
108                 callback();
109             }
110         }
111 
112     }(error_message, options.error_callback);
113 
114     return options;
115 };

117 /**
118  *  Another error wrapper, this is a clone of the backbone one.
119  **/
120 com.concertsoundorganizer.helpers.backboneWrapError = function(onError, model, options) {
121    return function(resp) {
122      if (onError) {
123        onError(model, resp, options);
124      } else {
125        model.trigger('error', model, resp, options);
126      }
127    };
128  };
