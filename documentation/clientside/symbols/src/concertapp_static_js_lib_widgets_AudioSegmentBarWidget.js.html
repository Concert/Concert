  7 /**
  8  *  A single audio segment bar at the top of the OverviewWaveformPanel.
  9  *  @class
 10  *  @extends    Widget
 11  **/
 12 var AudioSegmentBarWidget = Widget.extend({
 13     initialize: function() {
 14         Widget.prototype.initialize.call(this);
 15 
 16         var params = this.options;
 17         
 18         /**
 19          *  The right position of our el.  Cached so others can see.
 20          **/
 21         this.rightPx = null;
 22         
 23         /**
 24          *  The left position of our el.  Cached so others can see.
 25          **/
 26         this.leftPx = null;
 27 
 28         _.bindAll(this, "render");
 29     },
 30 
 31     render: function() {
 32         Widget.prototype.render.call(this);
 33         
 34         /* Our audio segment */
 35         var segment = this.model;
 36         
 37         /* Our width in px is the segment's duration multiplied by the panel's
 38         resolution (in px/second) */
 39         var width = (segment.get('end') - segment.get('beginning')) * this.panel.get_resolution();
 40         
 41         /* Our left is our start time multiplied by the panel's resolution */
 42         var left = segment.get('beginning') * this.panel.get_resolution();
 43         /* Plus one for border, and one for highlight.  
 44         TODO: Figure out how to do this with CSS. */
 45         left+=2;
 46         
 47         var el = $(this.el);
 48 
 49         el.css({
 50             width: width, 
 51             left: left 
 52         });
 53         
 54         /* Now we're rendered, can cache leftPx and rightPx property */
 55         this.leftPx = left;
 56         this.rightPx = left + el.width();
 57         
 58         return this;
 59     }
 60 });
