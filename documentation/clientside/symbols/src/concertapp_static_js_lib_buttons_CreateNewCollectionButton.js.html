  7 /**
  8  *  The button that will allow the user to create a new collection.
  9  *  @class
 10  **/
 11 function CreateNewCollectionButton(params) {
 12     if(params) {
 13         this.init(params);
 14     }
 15 }
 16 CreateNewCollectionButton.prototype = new Button();
 17 
 18 /**
 19  *  @constructor
 20  **/
 21 CreateNewCollectionButton.prototype.init = function(params) {
 22     Button.prototype.init.call(this, params);
 23     
 24     /* The panel we are on.  Need to access this to reset the form (for now this is
 25         good enough, but TODO: in the future, the CreateJoinCollectionPanel should
 26         be watching the results collection and calling the resetForm method 
 27         automatically when the collection changes.)
 28         */
 29     var panel = params.panel;
 30     if(typeof(panel) == 'undefined') {
 31         throw new Error('params.panel is undefined');
 32     }
 33     this.panel = panel;
 34     
 35     
 36     /* The name that the user has entered for the new collection */
 37     var newCollectionName = params.newCollectionName;
 38     if(typeof(newCollectionName) == 'undefined') {
 39         throw new Error('params.newCollectionName is undefined');
 40     }
 41     this.newCollectionName = newCollectionName;
 42     
 43     
 44 };
 45 
 46 /**
 47  *  This is what will be executed when the user clicks the button, deciding to 
 48  *  create a new collection.
 49  **/
 50 CreateNewCollectionButton.prototype.click = function() {
 51    /* Call corresponding method on page */
 52    this.panel.page.createNewCollection(this.newCollectionName);   
 53    /* reset form */
 54    this.panel.resetForm();
 55 };
