121     /**
122      *  Ensure that given audio models are shown as selected on the UI.  Takes
123      *  arrays in case we need to select multiple files/segments in the future.
124      *
125      *  @param  {Array}     params.files    -   The files selected.
126      *  @param  {Array}     params.segments -   The selected segments
127      **/
128     select_audio: function(params) {
129         /* Tell model manager so we can maintain list of instances */
130         this.modelManager.select_audio(params);
131         
132         /* Clear any highlights */
133         this.clear_waveform_highlight();
134         
135         
136         /* Determine what audio was selected */
137         var files = params.files;
138         var segments = params.segments;
139         files || (files = []);
140         segments || (segments = []);
141         
142         /* If one segment was selected */
143         if(segments.length == 1) {
144             /* Selecting an audio segment */
145             this.select_audio_segment(segments[0]);
146         }
147         /* If one file was selected */
148         else if(files.length == 1) {
149             /* Selecting a file */
150             this.select_audio_file(files[0]);
151         }
152         /* If files and segments were selected */
153         else if(files.length && segments.length) {
154             throw new Error('Not implemented multiple selection.');
155         }
156         /* If more than one file was selected */
157         else if(files.length > 1) {
158             throw new Error('Not implemented selecting multiple files');
159         }
160         /* If more than one segment was selected */
161         else if(segments.length > 1) {
162             throw new Error('Not implemented selecting multiple segments');
163         }
164         else {
165             throw new Error('Invalid parameters for select_audio');
166         }
167     },

304     /**
305      *  This will load an audio file into the audio player, and fire the callback
306      *  when complete.
307      *
308      *  @param  {AudioFile}    audioFile    -   the file to load.
309      *  @param  {Function}      callback    -   the function to call when done loading audio file.
310      **/
311     _load_audio_file: function(audioFile, callback) {
312         var audio = this.audio;
313 
314         /* The proper audio source for this browser */
315         var audiosrc = audioFile.get_audio_src(this.audioType);
316         
317         /* If this is a new audio file */
318         var newAudioSrc = !(audio.src.search(audiosrc) > 0);
319         
320         if(newAudioSrc) {
321             /* when the file is done loading */
322             $(audio).one('canplaythrough', callback);
323         
324             this.audio.src = audiosrc;
325         }
326         else {
327             callback();
328         }        
329     },

331     /**
332      *  When a user presses the space bar
333      **/
334     _handle_space_bar: function() {
335         /* Get HTML5 audio object */
336         var audio = this.audio;
337             
338         if(audio.paused) {
339             audio.play();
340         }
341         else {
342             audio.pause();
343         }

347     /**
348      *  This is called when a section of a waveform is selected.  It will start
349      *  an interval that will loop a single section of audio.
350      *
351      *  @param  {Number}    startTime    -  The beginning of the loop.
352      *  @param  {Number}    endTime     -   The end of the loop
353      **/
354     _start_audio_loop: function(startTime, endTime) {
355         if(this.audioLoopEnabled == false) {
356             var audio = this.audio;
357         
358             /* This function will be called when a timeupdate event occurs. */
359             var audioLoopTimeUpdateCallback = function(startTime, endTime) {
360                 return function(e) {
361                     var currentTime = audio.currentTime;
362                     if(currentTime < startTime || currentTime > endTime) {
363                         this.currentTime = startTime;
364                     }
365                 };
366             }(startTime, endTime);
367             /* Save so we can unbind later */
368             this.audioLoopTimeUpdateCallback = audioLoopTimeUpdateCallback;
369         
370             /* Start audio at beginning of loop */
371             if(audio.currentTime < startTime || audio.currentTime > endTime) {
372                 audio.currentTime = startTime;
373             }
374         
375             /* When audio loop changes time */
376             $(audio).bind('timeupdate', this.audioLoopTimeUpdateCallback); 
377             this.audioLoopEnabled = true;
378         }
379     }, 
380     
381     /**
382      *  This is called internally the audio loop is to be turned off
383      **/
384     _clear_audio_loop: function() {
385         if(this.audioLoopEnabled == true) {
386             $(this.audio).unbind('timeupdate', this.audioLoopTimeUpdateCallback);
387             this.audioLoopEnabled = false;
388         }
389     },
