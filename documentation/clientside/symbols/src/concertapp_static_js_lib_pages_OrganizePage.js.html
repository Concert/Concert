  7 /**
  8  *  The panels and widgets for the organize page.
  9  *	@class
 10  *  @extends    LoggedInPage
 11  **/
 12 var OrganizePage = LoggedInPage.extend(
 13 	/**
 14 	 *	@scope	OrganizePage.prototype
 15 	 **/
 16 {
 17     
 18     _initializeModelManager: function(params) {
 19         return new OrganizePageModelManager(params);
 20     }, 
 21     _initializeViews: function() {
 22         LoggedInPage.prototype._initializeViews.call(this);
 23         
 24         var modelManager = this.modelManager;
 25         
 26         /* This is our HTML5 audio player */
 27         var audio = new Audio();
 28         audio.autoplay = false;
 29         audio.preload = 'auto';
 30         this.audio = audio;
 31         
 32         /* The callback function for an audio loop */
 33         this.audioLoopTimeUpdateCallback = null;
 34         
 35         
 36         /* This is the type of audio file we will use */
 37         this.audioType = com.concertsoundorganizer.compatibility.audioType;
 38         
 39         
 40         /*  Create waveform overview panel */
 41         this.overviewPanel = new OverviewWaveformPanel({
 42             page: this, 
 43             el: $('#overview_waveform_panel'),
 44         });
 45 
 46         /* Create waveform detail panel */
 47         this.detailPanel = new DetailWaveformPanel({
 48             page: this, 
 49             el: $('#detail_waveform_panel'),
 50         });
 51 
 52 
 53         /* Create the audio list panel */    
 54         this.audioListPanel = new AudioListPanel({
 55             page: this, 
 56             el: $('#audio_list_panel'),
 57             files: modelManager.collectionAudioFiles,
 58             segments: modelManager.collectionAudioSegments,
 59             modelManager: modelManager, 
 60         });
 61         
 62         
 63         /* When the space button is pressed, pause/play our audio */
 64         $(window).bind('keydown', function(me) {
 65             return function(e) {
 66                 /* If this was a space, not from an input element */
 67                 if(e.keyCode == 32 && !$(e.srcElement).is('input')) {
 68                     e.preventDefault();
 69                     e.stopPropagation();
 70                     
 71                     me.play_pause();
 72                 }
 73             };
 74         }(this));
 75         
 76     }, 
 77     
 78     /**
 79      *  When a user selects some audio from the audio list.
 80      *
 81      *  @param  {Array}     params.files    -   The files selected.
 82      *  @param  {Array}     params.segments -   The selected segments
 83      **/
 84     select_audio: function(params) {
 85         /* Tell model manager so we can maintain list of instances */
 86         this.modelManager.select_audio(params);
 87         
 88         /* Pause audio if it is currently playing */
 89         this.pause();
 90         /* Clear any highlights/loops */
 91         this.clear_audio_loop();
 92         this.clear_waveform_highlight();
 93         
 94         
 95         /* Determine what audio was selected */
 96         var files = params.files;
 97         var segments = params.segments;
 98         files || (files = []);
 99         segments || (segments = []);
100         
101         /* If one segment was selected */
102         if(segments.length == 1) {
103             /* Selecting an audio segment */
104             this.select_audio_segment(segments[0]);
105         }
106         /* If one file was selected */
107         else if(files.length == 1) {
108             /* Selecting a file */
109             this.select_audio_file(files[0]);
110         }
111         /* If files and segments were selected */
112         else if(files.length && segments.length) {
113             throw new Error('Not implemented multiple selection.');
114         }
115         /* If more than one file was selected */
116         else if(files.length > 1) {
117             throw new Error('Not implemented selecting multiple files');
118         }
119         /* If more than one segment was selected */
120         else if(segments.length > 1) {
121             throw new Error('Not implemented selecting multiple segments');
122         }
123         else {
124             throw new Error('Invalid parameters for select_audio');
125         }
126     }, 
127     
128     /**
129      *  When a user has selected a single audio file.
130      *
131      *  @param  {AudioFile}    selectedAudioFile    -   The AudioFile that was
132      *  selected.
133      **/
134     select_audio_file: function(selectedAudioFile) {
135         /* This is where loading notification should be */
136         
137         /* Load audio file */
138         this._load_audio_file(selectedAudioFile, function(me, selectedAudioFile) {
139             /* when complete, notify panel */
140             return function(){
141                 me.overviewPanel.audio_file_selected(selectedAudioFile);
142                 me.detailPanel.audio_file_selected(selectedAudioFile);
143             };
144         }(this, selectedAudioFile));
145     },
146     
147     /**
148      *  When a user has selected a single audio segment
149      *
150      *  @param  {AudioSegment}    selectedAudioSegment    - The AudioSegment 
151      *  that was selected.
152      **/
153     select_audio_segment: function(selectedAudioSegment) {
154         /* Load the audio segment's file into the audio player */
155         this._load_audio_file(selectedAudioSegment.get('audioFile'), function(me, selectedAudioSegment) {
156             return function() {
157                 /* when complete, notify panels.  They will notify their 
158                 highlighters */
159                 me.overviewPanel.audio_segment_selected(selectedAudioSegment);
160                 me.detailPanel.audio_segment_selected(selectedAudioSegment);
161                 
162                 /* Start audio loop */
163                 me.start_audio_loop(
164                     selectedAudioSegment.get('beginning'),
165                     selectedAudioSegment.get('end')
166                 );
167             };
168         }(this, selectedAudioSegment))
169         
170     }, 
171     
172     /**
173      *  This will load an audio file into the audio player, and fire the callback
174      *  when complete.
175      *
176      *  @param  {AudioFile}    audioFile    -   the file to load.
177      **/
178     _load_audio_file: function(audioFile, callback) {
179         var audio = this.audio;
180         
181         /* when the file is done loading */
182         $(audio).one('canplaythrough', callback);
183         
184         
185         /* The proper audio source for this browser */
186         var audiosrc = audioFile.get_audio_src(this.audioType);
187         
188         this.audio.src = audiosrc;
189         
190     }, 
191     
192     /**
193      *  When a user presses the space bar
194      **/
195     play_pause: function() {
196         /* Get HTML5 audio object */
197         var audio = this.audio;
198             
199         if(audio.paused) {
200             this.play();
201         }
202         else {
203             this.pause();
204         }
205     }, 
206     
207     play: function() {
208         this.audio.play();
209         this.detailPanel.autoscrollBool = true;
210         
211     },
212     
213     pause: function() {
214         this.audio.pause();
215     },
216     
217     
218     
219     /**
220      *  This is called from a panel when an area of the waveform was highlighted by 
221      *  the user.
222      *
223      *  @param  {Number}    startTime    -  The time (in seconds) of highlight start
224      *  @param  {Number}    endTime    -    The time of the highlight end.
225      *  @param  {Panel}    panel    -   The panel that triggered the highlight
226      **/
227     waveform_highlighted: function(startTime, endTime, panel) {
228         /* Start audio loop */
229         this.start_audio_loop(startTime, endTime);
230         
231         /* If this highlight was from the detail panel */
232         if(panel instanceof DetailWaveformPanel) {
233             /* Tell overview panel */
234             this.overviewPanel.highlight_waveform(startTime, endTime);
235         }
236         else if(panel instanceof OverviewWaveformPanel) {
237             /* Tell detail panel */
238             this.detailPanel.highlight_waveform(startTime, endTime);
239         }
240         else {
241             throw new Error('Panel argument is invalid.');
242         }
243     },
244     
245     /**
246      *  This is called when a waveform is cleared.
247      *
248      *  @param  {Panel}    panel    -   The panel that cleared the highlight.
249      **/
250     waveform_highlight_cleared: function(panel) {
251         this.clear_audio_loop();
252         
253         if(panel instanceof DetailWaveformPanel) {
254             /* Tell overview panel */
255             this.overviewPanel.clear_waveform_highlight();
256         }
257         else if(panel instanceof OverviewWaveformPanel) {
258             this.detailPanel.clear_waveform_highlight();
259         }
260         else {
261             throw new Error('Panel argument is invalid')
262         }
263     }, 
264     
265     /**
266      *  This is called when a waveform should be cleared.
267      **/
268     clear_waveform_highlight: function() {
269         this.detailPanel.clear_waveform_highlight();
270         this.overviewPanel.clear_waveform_highlight();
271     }, 
272     
273     /**
274      *  This is called when a section of a waveform is selected.  It will start
275      *  an interval that will loop a single section of audio.
276      *
277      *  @param  {Number}    startTime    -  The beginning of the loop.
278      *  @param  {Number}    endTime     -   The end of the loop
279      **/
280     start_audio_loop: function(startTime, endTime) {
281         var audio = this.audio;
282                 
283         /* This function will be called when a timeupdate event occurs. */
284         var audioLoopTimeUpdateCallback = function(startTime, endTime) {
285             return function(e) {
286                 var currentTime = this.currentTime;
287                 
288                 if(currentTime < startTime || currentTime > endTime) {
289                     this.currentTime = startTime;
290                 }
291             };
292         }(startTime, endTime);
293         /* Save so we can unbind later */
294         this.audioLoopTimeUpdateCallback = audioLoopTimeUpdateCallback;
295         
296         /* Start audio at beginning of loop */
297         audio.currentTime = startTime;
298         
299         /* When audio loop changes time */
300         $(audio).bind('timeupdate', audioLoopTimeUpdateCallback);
301         
302     }, 
303     
304     /**
305      *  This is called when a highlight is cleared.
306      **/
307     clear_audio_loop: function() {
308         $(this.audio).unbind('timeupdate', this.audioLoopTimeUpdateCallback);
309     }, 
310     
311     move_audio: function(seconds) {
312         this.audio.currentTime = seconds;
313     }, 
314 });
