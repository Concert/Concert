  7 /**
  8  *  An audio file object.
  9  *  @class
 10  *  @extends    ConcertBackboneModel
 11  **/
 12 var AudioFile = ConcertBackboneModel.extend(
 13 	/**
 14 	 *	@scope	AudioFile.prototype
 15 	 **/
 16 {
 17     foreignKeyAttributes: function() {
 18         return [
 19             {
 20                 attr: 'uploader', 
 21                 model: User 
 22             },
 23             {
 24                 attr: 'collection', 
 25                 model: Collection
 26             }
 27         ];
 28     }, 
 29     oneToManyAttributes: function() {
 30         return [
 31             {
 32                 attr: 'segments', 
 33                 collectionType: AudioSegmentSet,
 34                 comparator: function(segment) {
 35                     return segment.get('beginning');
 36                 },  
 37             },
 38             {
 39                 attr: 'events', 
 40                 collectionType: EventSet, 
 41             }
 42         ];
 43     }, 
 44     name: 'audiofile', 
 45     
 46     /**
 47      *  Returns the path to the audio file specified by type.
 48      *
 49      *  @param  {String: ogg|mp3}    type    -   The type of audio file
 50      **/
 51     get_audio_src: function(type) {
 52         var id = this.get('id');
 53         
 54         if(id) {
 55             return '/media/audio/'+id+'.'+type;
 56         }
 57         else {
 58             return null;
 59         }
 60     }, 
 61     
 62     /**
 63      *  Returns the path to the waveform image specified by zoom_level
 64      *
 65      *  @param  {Number}    zoom_level    - The zoom level for this waveform image.
 66      **/
 67     get_waveform_src: function(zoom_level) {
 68         var id = this.get('id');
 69         
 70         if(id) {
 71             return '/media/waveforms/'+zoom_level+'/'+id+'.png';
 72         }
 73         else {
 74             return null;
 75         }
 76     }, 
 77     
 78 });
 79

 81  *  A set of audio file objects

 84  **/
 85 var AudioFileSet = ConcertBackboneCollection.extend(
 86     /**
 87 	 *	@scope	AudioFile.prototype
 88 	 **/
 89 	
 90 {
 91     model: AudioFile
 92 });
