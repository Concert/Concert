  7 /**
  8  *  Panel that displays larger waveform on organize page.
  9  *  @class
 10  *  @extends    WaveformPanel
 11  **/
 12 var DetailWaveformPanel = WaveformPanel.extend(
 13 	/**
 14 	 *	@scope	DetailWaveformPanel.prototype
 15 	 **/
 16 {    
 17     initialize: function() {
 18         WaveformPanel.prototype.initialize.call(this);
 19 
 20         var params = this.options;
 21         
 22         
 23         /* The template for the top of the panel when a file is selected */
 24         var topFileTemplate = $('#detail_waveform_top_file_template');
 25         if(typeof(topFileTemplate) == 'undefined') {
 26             throw new Error('$(\'#detail_waveform_top_file_template\') is undefined');
 27         }
 28         else if(topFileTemplate.length == 0) {
 29             throw new Error('topFileTemplate not found');
 30         }
 31         this.topFileTemplate = topFileTemplate;
 32         
 33         /* Template for top of panel when segment is selected */
 34         var topSegmentTemplate = $('#detail_waveform_top_segment_template');
 35         if(typeof(topSegmentTemplate) == 'undefined') {
 36             throw new Error('$(\'#detail_waveform_top_segment_template\') is undefined');
 37         }
 38         else if(topSegmentTemplate.length == 0) {
 39             throw new Error('topSegmentTemplate not found');
 40         }
 41         this.topSegmentTemplate = topSegmentTemplate;
 42         
 43         /* The container for the top of the panel */
 44         var topContainer = $('#detail_waveform_panel_top');
 45         if(typeof(topContainer) == 'undefined') {
 46             throw new Error('$(\'#detail_waveform_panel_top\') is undefined');
 47         }
 48         else if(topContainer.length == 0) {
 49             throw new Error('topContainer not found');
 50         }
 51         this.topContainer = topContainer;
 52                 
 53         var timecodeContainerElement = $('#detail_waveform_panel_timecode');
 54         if(typeof(timecodeContainerElement) == 'undefined') {
 55             throw new Error('$(\'#detail_waveform_panel_timecode\') is undefined');
 56         }
 57         else if(timecodeContainerElement.length == 0) {
 58             throw new Error('timecodeContainerElement not found');
 59         }
 60         this.timecodeContainerElement = timecodeContainerElement;
 61         
 62         /* Instantiate widget for timecode */
 63         var timecodeComponent = new DetailWaveformTimecodeComponent({
 64             el: timecodeContainerElement, 
 65             panel: this, 
 66             audio: this.page.audio
 67         });
 68         this.timecodeComponent = timecodeComponent;
 69         
 70         /* Instantiate component for playhead */
 71         var playheadComponent = new DetailWaveformPlayheadComponent({
 72             el: this.playheadContainerElement,
 73             panel: this,
 74             audio: this.page.audio
 75         });
 76         this.playheadComponent = playheadComponent;
 77         
 78         
 79         var highlightContainerElement = $('#detail_waveform_panel_highlight_container');
 80         if(typeof(highlightContainerElement) == 'undefined') {
 81             throw new Error('$(\'#detail_waveform_panel_highlight_container\') is undefined');
 82         }
 83         else if(highlightContainerElement.length == 0) {
 84             throw new Error('highlightContainerElement not found');
 85         }
 86         this.highlightContainerElement = highlightContainerElement;
 87         
 88         /* a highlighter component so we can highlight things */
 89         var highlighter = new DetailWaveformHighlighterComponent({
 90             el: highlightContainerElement, 
 91             panel: this, 
 92         });
 93         this.highlighter = highlighter;
 94         
 95         var waveformView = $('#detail_waveform_panel_view');
 96         if(typeof(waveformView) == 'undefined') {
 97             throw new Error('$(\'#detail_waveform_panel_view) is undefined');
 98         }
 99         else if(waveformView.length == 0) {
100             throw new Error('waveformView not found');
101         }
102         this.waveformView = waveformView;
103         
104         var autoscrollBool = true;
105         this.autoscrollBool = autoscrollBool;
106         
107         this.waveformView.bind('scrollstart', function(me) {
108             return function() {
109                 me.handle_scroll();
110             };
111         }(this));
112         
113         this.waveformView.bind('scrollstop', function(me) {
114             return function() {
115                 me.handle_scroll_stop();
116             }
117         }(this));
118         
119         this.waveformView.bind('click', function(me) {
120             return function(e) {
121                 me.handle_click(get_event_x(e));
122             };
123         }(this));            
124     },
125     /**
126      *  Called from page when an audio file has been selected.
127      *
128      *  @param  {AudioFile}    selectedAudioFile    -   The selected file.
129      **/
130     audio_file_selected: function(selectedAudioFile) {
131         WaveformPanel.prototype.audio_file_selected.call(this, selectedAudioFile);
132         
133         /* Load top content with audio file information */
134         this.topContainer.html(
135             this.topFileTemplate.tmpl(selectedAudioFile.toJSON())
136         );
137         
138         /* Create editable text component to handle name change */
139         this.topNameComponent = new EditableModelTextComponent({
140             panel: this, 
141             model: selectedAudioFile, 
142             attr: 'name',
143             el: $('#detail_waveform_selected_name_container')
144         });
145         
146         /* Load waveform image */
147         this._load_waveform_image(selectedAudioFile.get_waveform_src(10), function(me, selectedAudioFile) {
148             /* and when done */
149             return function() {
150                 /* Draw timecode */
151                 me.timecodeComponent.audio_file_selected(selectedAudioFile);
152                 
153                 /* Set up highlighter */
154                 me.highlighter.audio_file_selected(selectedAudioFile);
155             };            
156         }(this, selectedAudioFile));
157     }, 
158     
159     /**
160      *  Called from page when audio segment has been selected.
161      *
162      *  @param  {AudioSegment}    selectedAudioSegment    - The selected segment.
163      **/
164     audio_segment_selected: function(selectedAudioSegment) {
165         WaveformPanel.prototype.audio_segment_selected.call(this, selectedAudioSegment);
166         
167         /* Load top of panel with audio segment information */
168         this.topContainer.html(
169             this.topSegmentTemplate.tmpl(selectedAudioSegment.toJSON())
170         );
171 
172         /* Create editable text component to handle name change */
173         this.topNameComponent = new EditableModelTextComponent({
174             panel: this, 
175             model: selectedAudioFile, 
176             attr: 'name',
177             el: $('#detail_waveform_selected_name_container')
178         });
179         
180         /* Load waveform image */
181         this._load_waveform_image(
182             selectedAudioSegment.get('audioFile').get_waveform_src(10),
183             function(me, selectedAudioSegment) {
184                 return function() {
185                     /* Draw timecode */
186                     me.timecodeComponent.audio_segment_selected(selectedAudioSegment);
187                     
188                     me.highlighter.audio_segment_selected(selectedAudioSegment);
189                 };
190             }(this, selectedAudioSegment)
191         );
192         
193     }, 
194     
195     
196     /**
197      *  Called from highlight when an area of the waveform is highlighted.
198      *
199      *  @param  {Number}    startTime    -  The time (in seconds) of highlight start
200      *  @param  {Number}    endTime    -    The time of the highlight end.
201      **/
202     waveform_highlighted: function(startTime, endTime) {
203         this.autoscrollBool = true;
204         
205         /* Tell page about our highlight */
206         this.page.waveform_highlighted(startTime, endTime, this);
207     }, 
208     
209     
210     /**
211      *  Determines if the playhead is in the current waveform view
212      **/
213     playhead_in_view: function() {
214         if (((this.waveformView.scrollLeft()) <= this.playheadComponent.position()) && 
215         ((this.waveformView.scrollLeft() + 815) >= this.playheadComponent.position())) {
216             return true;
217         }
218         else {
219             return false;
220         }
221     },
222     
223     /**
224      *  if the playhead is at the end of the view && AUTOSCROLLING ON 
225      *  scroll to position of playhead
226      **/
227     playhead_moved: function(leftPx) {        
228         if ((this.autoscrollBool && leftPx >= (815 + this.waveformView.scrollLeft())) ||
229             (this.autoscrollBool && leftPx < this.waveformView.scrollLeft())) {
230             this.autoscroll(leftPx);
231         }
232 
233     }, 
234     
235     /**
236      *  Handles playhead behavior on scroll
237      **/
238     handle_scroll: function() {
239         this.autoscrollBool = false;
240     }, 
241 
242     /**
243     *  Handles playhead behavior on a scroll stop
244     **/    
245     handle_scroll_stop: function() {
246         if (!this.autoscrollBool && this.playhead_in_view()) {
247             this.autoscrollBool = true;
248         }
249         else {
250             this.autoscrollBool = false;
251         }
252     }, 
253         
254     /**
255      *  moves the waveform view to leftPx
256      **/
257     autoscroll: function(leftPx) {
258         this.waveformView.animate({scrollLeft: leftPx}, 600, "easeOutExpo");
259     },
260     
261     
262     handle_click: function(left) {
263         //update audio's currentTime to location clicked
264         var leftPx = left + this.waveformView.scrollLeft();
265         var seconds = leftPx/10;
266         this.page.move_audio(seconds);
267     }, 
268     
269     /**
270      *  The resolution of the waveform image (in pixels/second)
271      **/
272     get_resolution: function() {
273         return 10;
274     }, 
275 });
