  7 /**
  8  *  The widget that displays the timecode underneath the detail waveform.
  9  *  @class
 10  *  @extends    Component
 11  **/
 12 var DetailWaveformTimecodeComponent = Component.extend(
 13 	/**
 14 	 *	@scope	DetailWaveformTimecodeComponent.prototype
 15 	 **/
 16 {
 17     initialize: function() {
 18         Component.prototype.initialize.call(this);
 19 
 20         var params = this.options;
 21         
 22         /* The HTML5 audio element that we will watch */
 23         var audio = params.audio;
 24         if(typeof(audio) == 'undefined') {
 25             throw new Error('params.audio is undefined');
 26         }
 27         this.audio = audio;
 28         
 29         /* The duration of the currently selected audio file (or segment parent) */
 30         this.fileDuration = null;
 31     },
 32     
 33     /**
 34      *  Called from panel when audio file was selected.
 35      *
 36      *  @param  {AudioFile}    selectedAudioFile    -   selected audio file
 37      **/
 38     audio_file_selected: function(selectedAudioFile) {
 39         /* Save duration of audio */
 40         this.fileDuration = selectedAudioFile.get('duration');
 41         
 42         /* render */
 43         this.render();
 44         
 45     }, 
 46     
 47     /**
 48      *  Called from panel when audio segment was selected.
 49      *
 50      *  @param  {AudioSegment}    selectedAudioSegment    - selected segment
 51      **/
 52     audio_segment_selected: function(selectedAudioSegment) {
 53         /* Save duration of audio file associated with this segment */
 54         this.fileDuration = selectedAudioSegment.get('audioFile').get('duration');
 55         
 56         /* Render */
 57         this.render();
 58     }, 
 59     
 60     /**
 61      *  Called internally when timecode is to be redrawn.
 62      **/
 63     render: function() {
 64         Component.prototype.render.call(this);
 65         var el = this.el;
 66         
 67         /* Pixels per second currently is 10 because we only have one zoom level */
 68         var pxPerSecond = 10;
 69         
 70         /* Clear canvas */
 71         this.el.empty();
 72         
 73         var duration = this.fileDuration;
 74         
 75         /* Draw timecode with canvas */
 76         $g().size(pxPerSecond*duration, el.height())
 77             .add(function(pxPerSecond, duration) {
 78                 return function(ctx, canvas) {
 79                     var height = canvas.height;
 80                     var width = canvas.width;
 81                     
 82                     /* Begin our cursor at the top left corner of the canvas */
 83                     var cursor = {
 84                         x: 0, 
 85                         y: 0
 86                     };
 87                     
 88                     /* We will draw black 1px lines for now */
 89                     ctx.strokeStyle = 'black';
 90                     ctx.lineCap = 'square';
 91                     ctx.lineWidth = 1.0;
 92 
 93                     ctx.beginPath();
 94                     
 95                     /* Place long marker every 10 seconds */
 96                     for(var i = 0; i < duration; i+=10) {
 97                         /* Draw marker for this second */
 98                         
 99                         /* Move cursor horizontally */
100                         cursor.x = i*pxPerSecond;
101                         cursor.y = 0;
102                         ctx.moveTo(cursor.x, cursor.y);
103                         
104                         /* Draw a vertical line at current point */
105                         ctx.lineTo(cursor.x, cursor.y+(height*0.50));
106                         ctx.stroke();
107                     }
108                     
109                     /* Place short marker every 10 seconds (on multiples of 5) */
110                     for(var i = 5; i < duration; i+=5) {
111                         cursor.x = i*pxPerSecond;
112                         cursor.y = 0;
113                         ctx.moveTo(cursor.x, cursor.y);
114                         
115                         ctx.lineTo(cursor.x, cursor.y+(height * 0.25));
116                         ctx.stroke();
117                     }
118                      /* Place short marker every 1 seconds */
119                     for(var i = 0; i < duration; i+=1) {
120                         cursor.x = i*pxPerSecond;
121                         cursor.y = 0;
122                         ctx.moveTo(cursor.x, cursor.y);
123                         
124                         ctx.lineTo(cursor.x, cursor.y+(height * 0.125));
125                         ctx.stroke();
126                     }
127                     ctx.closePath();
128                 };
129             }(pxPerSecond, duration))
130             .place(el).draw();
131         
132         
133         return this;
134     }
135 });
