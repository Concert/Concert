  7 /**
  8  *  The panel on the top of every page.
  9  *	@class
 10  *  @param  {jQuery HTMLSelectElement}      params.collectionSelector - The
 11  *                                          select which chooses which collection to
 12  *                                          organize.
 13  *  @param  {UploadLinkLargeIconButton}     params.uploadButton
 14  *
 15  **/
 16 var GlobalOptionsPanel = Panel.extend(
 17 	/**
 18 	 *	@scope	GlobalOptionsPanel.prototype
 19 	 **/
 20 {
 21     initialize: function() {
 22         Panel.prototype.initialize.call(this);
 23 
 24         var params = this.options;
 25 
 26         var contents = this.contents;
 27 
 28         /* Make sure parameter was passed in */
 29         var collectionSelector = contents.find('.collection_selector');
 30         if(typeof(collectionSelector) == 'undefined' || collectionSelector.length == 0) {
 31             throw new Error('collectionSelector not found');
 32         }
 33         /* Save as member */
 34         this.collectionSelector = collectionSelector;
 35 
 36 
 37         /* Get upload button */
 38         var uploadButtonContainer = contents.find('#upload_button');
 39         /* If this page has an upload button */
 40         if(uploadButtonContainer.length) {
 41             /* Create UploadLinkLargeIconButton object */
 42             this.uploadButton = new UploadLinkLargeIconButton({
 43                 container: uploadButtonContainer, 
 44             });        
 45         }
 46 
 47         /* get settings button */
 48         var settingsButtonContainer = contents.find('#settings_button');
 49         /* if this page has a settings button */
 50         if(settingsButtonContainer.length) {
 51             /* Create settings Button */
 52             this.settingsButton = new SettingsLinkLargeIconButton({
 53                 container: settingsButtonContainer, 
 54             });
 55         }
 56 
 57         var userMemberCollections = params.userMemberCollections;
 58         if(typeof(userMemberCollections) == 'undefined') {
 59             throw new Error('params.userMemberCollections is undefined');
 60         }
 61         this.userMemberCollections = userMemberCollections;
 62 
 63         
 64 
 65 
 66         /** INitialize behavior for collection selector */
 67         collectionSelector.bind('change', function(e) {
 68             var collection_id = $(this).val();
 69 
 70             window.location = '/organize/collection/'+collection_id;
 71         });
 72 
 73         /* This is the template for the collection dropdown */
 74         var collectionSelectorOptionsTemplate = $('#collection_selector_options_template');
 75         if(typeof(collectionSelectorOptionsTemplate) == 'undefined' || collectionSelectorOptionsTemplate.length == 0) {
 76             throw new Error('collectionSelectorOptionsTemplate not found');
 77         }
 78         this.collectionSelectorOptionsTemplate = collectionSelectorOptionsTemplate;
 79 
 80 
 81         
 82         _.bindAll(this, "render");
 83         /* Bind collection events to render */
 84         userMemberCollections.bind('refresh', this.render);
 85         userMemberCollections.bind('add', this.render);
 86         userMemberCollections.bind('remove', this.render);
 87     },
 88     
 89     /* To render this panel, just populate the dropdown */
 90     render: function() {
 91         
 92         /* If there are collections */
 93         if(this.userMemberCollections.length) {
 94             /* The dropdown should not be disabled.  TODO: This logic can be in the template. */
 95             if(this.collectionSelector.attr('disabled')) {
 96                 this.collectionSelector.removeAttr('disabled');
 97             }
 98             /* Populate dropdown */
 99             this.collectionSelector.html(
100                 this.collectionSelectorOptionsTemplate.tmpl({
101                     collections: this.userMemberCollections.toJSON(), 
102                 })
103             );            
104         }
105         else {
106             /* There are not collections */
107             
108             /* Disable dropdown */
109             this.collectionSelector.attr('disabled', 'true');
110             
111             /* TODO: Display notification, or some text in selector?  This logic can also be in the template. */
112         }
113 
114 
115         return this;
116     }
117 
118 });
