  7 /**
  8  *  The component that allows for highlighting functionality on a WaveformPanel.
  9  *  @class
 10  *  @extends    Component
 11  *  @throws     Highlight   -   When a section of the waveform is highlighted.
 12  **/
 13 var WaveformHighlighterComponent = Component.extend(
 14 	/**
 15 	 *	@scope	WaveformHighlighterComponent.prototype
 16 	 **/
 17 {
 18     initialize: function() {
 19         Component.prototype.initialize.call(this);
 20         
 21         var params = this.options;
 22         
 23         var el = this.el;
 24         
 25         
 26         /* Determine if highlight is currently disabled (incase someone messes 
 27         with template) */
 28         if(el.hasClass('disabled')) {
 29             this.disabled = true;
 30         }
 31         else {
 32             this.disabled = false;
 33         }
 34         
 35         /* If we are currently dragging a highlight */
 36         this.dragging = false;
 37         
 38         /* Where the last drag started (x-coordinate) */
 39         this.lastDragStartX = null;
 40         
 41         /* Where the last drag point was (x-coordinate) */
 42         this.lastDragEndX = null;
 43         
 44         /* Cache the duration of the current audio file (we will need it each time
 45         we highlight) */
 46         this.audioFileDuration = null;
 47     }, 
 48     
 49     _initializeElements: function() {
 50         Component.prototype._initializeElements.call(this);
 51         
 52         
 53         /* Get highlight element inside this container */
 54         var highlight = this.el.children('.highlight');
 55         if(typeof(highlight) == 'undefined') {
 56             throw new Error('this.el.children(\'.highlight\') is undefined');
 57         }
 58         else if(highlight.length == 0) {
 59             throw new Error('highlight not found');
 60         }
 61         this.highlight = highlight;
 62         
 63 
 64     },
 65     
 66     _initializeEvents: function(){
 67         Component.prototype._initializeEvents.call(this);
 68         
 69         /* The element that we are receiving the drag events from (should be defined
 70         in child classes) */
 71         var el = this.el;
 72         
 73         el.bind('mousedown', function(me) {
 74             return function(e) {
 75                 e.stopPropagation();
 76                 
 77                 me.startDrag(get_event_x(e));
 78             };
 79         }(this));
 80         
 81         el.bind('mousemove', function(me) {
 82             return function(e) {
 83                 e.stopPropagation();
 84                 
 85                 if(me.dragging) {
 86                     me.continueDrag(get_event_x(e));
 87                 }
 88             };
 89         }(this));
 90         
 91         el.bind('mouseup', function(me) {
 92             return function(e) {
 93                 e.stopPropagation();
 94                 
 95                 me.endDrag(get_event_x(e));
 96             };
 97         }(this));
 98     }, 
 99     
100     /**
101      *  When a drag is started.
102      *
103      *  @param  {Number}    x    -  The x-coordinate where the drag began.
104      **/
105     startDrag: function(x) {
106         /* Reset any old highlight */
107         this.reset();
108         /* Make highlight visible */
109         this.enable();
110         /* Save new starting point */
111         this.lastDragStartX = x;
112         /* We are now dragging */
113         this.dragging = true;
114     }, 
115     
116     /**
117      *  When a drag is continuing.  Called on every mousemove event.
118      *
119      *  @param  {Number}    x   -   The x-coordinate where the drag is currently.
120      **/
121     continueDrag: function(x) {
122         this.lastDragEndX = x;
123         
124         /* Draw highlight */
125         this.draw_highlight_px(this.lastDragStartX, this.lastDragEndX);
126     }, 
127     
128     /**
129      *  When a drag has stopped.
130      *
131      *  @param  {Number}    x    -  The x-coordinate where the drag has stopped.
132      **/
133     endDrag: function(x) {
134         this.lastDragEndX = x;
135         
136         this.dragging = false;
137         
138         /* Now determine what the time of the highlight was relative to the audio
139         file */
140         var dragStartX = this.lastDragStartX;
141         var dragEndX = x;
142         /* If this was just a click */
143         if(dragStartX == dragEndX) {
144             /* Don't do anything */
145             return;
146         }
147         var pxPerSecond = this.panel.get_resolution();
148         var startTime = Math.min(dragStartX, dragEndX)/pxPerSecond;
149         var endTime = Math.max(dragStartX, dragEndX)/pxPerSecond;
150         
151         /* Tell panel about highlight */
152         this.panel.waveform_highlighted(startTime, endTime);
153         
154     }, 
155     
156     /**
157      *  Called when a highlight is to be drawn on the waveform by pixels.  Takes
158      *  two arguments, each of which are x-coordinates relative to the highlight
159      *  container.  The order doesn't matter.
160      *
161      *  @param  {Number}    x    -  One side of the highlight
162      *  @param  {Number}    y    -  The other side.
163      **/
164     draw_highlight_px: function(x, y) {
165         /* If the values are left to right */
166         if(x < y) {
167             this.highlight.css({
168                 left: x+'px', 
169                 width: (y-x)+'px' 
170             });
171         }
172         /* If the drag is from right to left */
173         else if(x > y) {
174             this.highlight.css({
175                 left: y+'px', 
176                 width: (x-y)+'px'
177             });
178         }        
179     }, 
180     
181     /**
182      *  Called when a highlight is to be drawn on the waveform, but we only have
183      *  time information.  Takes two arguments, one for each side of the highlight.
184      *
185      *  @param  {Number}    t1  -   One side of the highlight
186      *  @param  {Number}    t2    - The other side.
187      **/
188     draw_highlight_sec: function(t1, t2) {
189         var duration = this.audioFileDuration;
190         var pxPerSecond = this.panel.get_resolution();
191         var x = (t1/duration)*(duration*pxPerSecond);
192         var y = (t2/duration)*(duration*pxPerSecond);
193         this.draw_highlight_px(x, y);
194     }, 
195     
196     /**
197      *  When an audio file has been selected.  Called from panel.
198      *
199      *  @param  {AudioFile}    selectedAudioFile    -   The audio file object.
200      **/
201     audio_file_selected: function(selectedAudioFile) {
202         
203         var audioFileDuration = selectedAudioFile.get('duration');
204         this.audioFileDuration = audioFileDuration;
205         
206         /* Set width of highlight container element properly */
207         this.el.css('width', audioFileDuration*this.panel.get_resolution()+'px');
208     }, 
209     
210     /**
211      *  When an audio segment has been selected.  Called from panel.
212      *
213      *  @param  {AudioSegment}    selectedAudioSegment    - The segment obj.
214      **/
215     audio_segment_selected: function(selectedAudioSegment) {
216         var audioFileDuration = selectedAudioSegment.get('audioFile').get('duration');
217         this.audioFileDuration = audioFileDuration;
218         
219         /* Set width of highlight container element */
220         this.el.css('width', audioFileDuration*this.panel.get_resolution()+'px');
221         
222         /* Draw highlight for this segment */
223         this.draw_highlight_sec(
224             selectedAudioSegment.get('beginning'),
225             selectedAudioSegment.get('end')
226         );
227         
228         /* show highlight */
229         this.enable();
230     }, 
231     
232     /**
233      *  When this highlight is being used.  Make sure it is visible on UI.
234      **/
235     enable: function() {
236         this.disabled = false;
237         this.highlight.removeClass('disabled');
238     }, 
239     
240     /**
241      *  When this highlight is not being used.  It will not be shown on the screen.
242      **/
243     disable: function() {
244         this.disabled = true;
245         this.highlight.addClass('disabled');            
246     }, 
247     
248     /**
249      *  When the highlight is to be reset.  This will reset the width to 0.
250      **/
251     reset: function() {
252         this.highlight.css({
253             width: '0px'
254         });
255         
256         this.panel.waveform_highlight_cleared();
257     }
258 
259 });
