  7 /**
  8  *  A Request that represents a Request resource on the server.
  9  *  @class
 10  *  @extends    ConcertBackboneModel
 11  **/ 
 12 var Request = ConcertBackboneModel.extend(
 13 	/**
 14 	 *	@scope	Request.prototype
 15 	 **/
 16 {
 17     
 18     foreignKeyAttributes: function() {
 19         return [
 20             {
 21                 attr: 'user', 
 22                 model: User
 23             },
 24             {
 25                 attr: 'collection', 
 26                 model: Collection 
 27             }
 28         ];
 29     },
 30     name: 'request',
 31     /**
 32      *  When a user would like to revoke the request to join a collection.
 33      **/
 34     revoke: function() {
 35         this._change_status_and_handle_error('r', 'Request could not be revoked');
 36     }, 
 37     /**
 38      *  When an administrator denies a request
 39      **/
 40     deny: function() {
 41         this._change_status_and_handle_error('d', 'Request was not denied');
 42     },
 43     /**
 44      *  When an administrator approves a request
 45      **/
 46     approve: function() {
 47         var previousStatus = this.get('status');
 48         var addedUserId = this.get('user').get('id');
 49         
 50         this.set({
 51             status: 'a'
 52         });
 53         
 54         /* Add user to collection */
 55         this.get('collection').get('users').add(this.get('user'));
 56 
 57         /* Save changes. */
 58         this.save(null, {
 59             error_message: 'Request could not be approved', 
 60             error_callback: function(me, previousStatus, addedUserId) {
 61                 return function() {
 62                     /* On error, put request back to previous state */
 63                     me.set({
 64                         status: previousStatus 
 65                     });
 66                     /* Remove user from collection */
 67                     me.get('collection').get('users').remove(addedUserId);
 68                 };
 69             }
 70         });
 71         
 72     }, 
 73     /**
 74      *  A helper function for changing the status of a request, and changing it back
 75      *  if an error occurs.
 76      **/
 77     _change_status_and_handle_error: function(newStatus, error_message) {
 78         var previousStatus = this.get('status');
 79         this.set({
 80             status: newStatus
 81         });
 82         this.save(null, {
 83             error_message: error_message, 
 84             error_callback: function(me, previousStatus) {
 85                 return function() {
 86                     me.set({
 87                         status: previousStatus
 88                     });
 89                 };
 90             }(this, previousStatus)
 91         });
 92     }
 93 });
 94 
 95 /**
 96  *  A set of Request objects.

 99  **/
100 var RequestSet = ConcertBackboneCollection.extend(
101 	/**
102 	 *	@scope	RequestSet.prototype
103 	 **/
104 {
105     model: Request, 
106 });
