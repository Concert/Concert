  7 /**
  8  *  @class  Panel that shows events on the organize page.
  9  *  @extends    Panel
 10  **/
 11 var EventsPanel = Panel.extend({
 12     initialize: function() {
 13         Panel.prototype.initialize.call(this);
 14 
 15         var params = this.options;
 16         
 17         /**
 18          *  Which widget class do we use for which event type.
 19          **/
 20         this.eventTypesToWidgetMap = {
 21             1: AudioSegmentCreatedEventWidget,
 22             2: AudioSegmentTaggedEventWidget,
 23             3: AudioFileUploadedEventWidget,
 24             4: JoinCollectionEventWidget,
 25             5: LeaveCollectionEventWidget,
 26             6: CreateCollectionEventWidget,
 27             7: RequestJoinCollectionEventWidget,
 28             8: RequestDeniedEventWidget,
 29             9: RequestRevokedEventWidget
 30         }
 31         
 32         /**
 33          *  Which template we will use for each event type.
 34          **/
 35         this.eventTypesToTemplateMap = {
 36             1: $('#audiosegmentcreatedevent_template'),
 37             2: $('#audiosegmenttaggedevent_template'),
 38             3: $('#audiofileuploadedevent_template'),
 39             4: $('#joincollectionevent_template'),
 40             5: $('#leavecollectionevent_template'),
 41             6: $('#createcollectionevent_template'),
 42             7: $('#requestjoincollectionevent_template'),
 43             8: $('#requestdeniedevent_template'),
 44             9: $('#requestrevokedevent_template')
 45         }
 46         
 47         /**
 48          *  The template used for rendering an event widget
 49          **/
 50         var eventWidgetTemplate = $('#event_template');
 51         if(typeof(eventWidgetTemplate) == 'undefined') {
 52             throw new Error('$(\'#event_template\') is undefined');
 53         }
 54         else if(eventWidgetTemplate.length == 0) {
 55             throw new Error('eventWidgetTemplate not found');
 56         }
 57         this.eventWidgetTemplate = eventWidgetTemplate;
 58         
 59 
 60         _.bindAll(this, "render");
 61     },
 62 
 63     /**
 64      *  Here we will render the list of events.  An argument is needed, the set of
 65      *  event models.
 66      *
 67      *  @param  {EventSet}    eventModels    The models to render
 68      **/
 69     render: function(eventModels) {
 70         Panel.prototype.render.call(this);
 71         
 72         var panel = this;
 73         var eventTypesToWidgetMap = this.eventTypesToWidgetMap;
 74         var eventTypesToTemplateMap = this.eventTypesToTemplateMap;
 75         var frag = document.createDocumentFragment();
 76         /* For each event model */
 77         eventModels.each(function(eventModel) {
 78             var eventType = eventModel.get('eventType');
 79             /* Proper widget for this event */
 80             var widgetClass = eventTypesToWidgetMap[eventType];
 81             var widgetTemplate = eventTypesToTemplateMap[eventType];
 82             
 83             /* Create widget */
 84             var widget = new widgetClass({
 85                 panel: panel, 
 86                 model: eventModel,
 87                 template: widgetTemplate 
 88             });
 89             
 90             frag.appendChild(widget.render().el);
 91             
 92             
 93         });
 94         
 95         this.contents.html(frag);
 96         
 97         return this;
 98     }
 99 });
