  7 /**
  8  *  This is the widget that displays the search results when searching for a
  9  *  collection.
 10  *  @class
 11  *  @extends Widget
 12  **/
 13 var CollectionSearchResultsWidget = Widget.extend(
 14 	/**
 15 	 *	@scope	CollectionSearchResultsWidget.prototype
 16 	 **/
 17 {
 18     initialize: function() {
 19         Widget.prototype.initialize.call(this);
 20 
 21         var params = this.options;
 22         
 23         var resultTemplate = $('#create_join_result_template');
 24         if(typeof(resultTemplate) == 'undefined' || resultTemplate.length == 0) {
 25             throw new Error('resultTemplate not found');
 26         }
 27         this.resultTemplate = resultTemplate;
 28                 
 29         var searchResults = params.searchResults;
 30         if(typeof(searchResults) == 'undefined') {
 31             throw new Error('params.searchResults is undefined');
 32         }
 33         this.searchResults = searchResults;
 34 
 35         
 36 
 37         /* Bind collection events to render */
 38         searchResults.bind('refresh', this.render);
 39     },
 40     render: function() {
 41         
 42         var collections = this.searchResults;
 43 
 44         /* Clear results */
 45         $(this.el).empty();
 46         
 47         var resultTemplate = this.resultTemplate;
 48         
 49         var frag = document.createDocumentFragment();
 50         
 51         var currentTerm = this.panel.currentTerm;
 52         var exact = this.panel.exactResult;
 53         
 54         /* If there were search results */
 55         if(collections.length) {
 56             /* For each element in the collection */
 57             collections.each(function(frag, resultTemplate, panel) {
 58                 return function(collection){
 59 
 60                     /* Create a CollectionSearchResult widget */
 61                     var widget = new CollectionSearchResultWidget({
 62                         template: resultTemplate, 
 63                         model: collection,
 64                         panel: panel, 
 65                     });
 66 
 67                     frag.appendChild(widget.el);
 68 
 69                 };
 70             }(frag, resultTemplate, this.panel));            
 71         }
 72         
 73         $(this.el).html(frag);
 74         return this;
 75     }
 76 });
