  8 /**
  9  *  This is the panel where the user can manage the collections he/she has requested
 10  *  to join.
 11  *  @class
 12  *  @extends    ManageTablePanel
 13  **/
 14 var ManageRequestsPanel = ManageTablePanel.extend(
 15 	/**
 16 	 *	@scope	ManageRequestsPanel.prototype
 17 	 **/
 18 {
 19     initialize: function() {
 20         ManageTablePanel.prototype.initialize.call(this);
 21 
 22         /* The manage admin collection widget template */
 23         var widgetTemplate = $('#manage_request_collection_template');
 24         if(typeof(widgetTemplate) == 'undefined' || widgetTemplate.length == 0) {
 25             throw new Error('widgetTemplate not found.');
 26         }
 27         this.widgetTemplate = widgetTemplate;
 28         
 29         /* The ManageAdminCollectionWidget class */
 30         this.widgetClass = ManageRequestWidget;
 31         
 32     }
 33 });
