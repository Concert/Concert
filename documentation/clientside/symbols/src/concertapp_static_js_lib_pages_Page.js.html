  7 /**
  8  *  This is the class that contains the functionality that needs to be run on
  9  *  every page.
 10  *	@class
 11  *  @extends    Backbone.Controller
 12  **/
 13 var Page = Backbone.Controller.extend(
 14 	/**
 15 	 *	@scope	Page.prototype
 16 	 **/
 17 {
 18     initialize: function(params) {
 19         /* Create dataset manager */
 20         var modelManager = this._initializeModelManager(_.extend(params, {
 21             page: this 
 22         }));
 23         this.modelManager = modelManager;
 24         com.concertsoundorganizer.modelManager = modelManager;
 25 
 26         /* Create views */
 27         this._initializeViews();
 28         
 29         /* Load data so views can update and such */
 30         this.modelManager._loadData();
 31         
 32         /* Initialize routes */
 33         this._initialize_routes();
 34         
 35         Backbone.history.start();
 36     }, 
 37     /**
 38      *  This method just defines which dataset manager to use.  Should be overridden
 39      *  in child classes for appropriate type.
 40      *
 41      **/
 42     _initializeModelManager: function(params) {
 43         return new ModelManager(params);
 44     }, 
 45     
 46     /**
 47      *  This method is called when views are to be created.  Should be overridden
 48      *  in child classes.
 49      **/
 50     _initializeViews: function() {
 51         return;
 52     }, 
 53     
 54     /**
 55      *  This method is called when routes are to be initialized.
 56      **/
 57     _initialize_routes: function() { 
 58         
 59         _.bindAll(this, this._default_route);
 60         this.route('', 'default', this._default_route);
 61         
 62         return;
 63     }, 
 64     
 65     /**
 66      *  This is the default route that will be executed.
 67      **/
 68     _default_route: function() {
 69         return;
 70     }, 
 71 });
