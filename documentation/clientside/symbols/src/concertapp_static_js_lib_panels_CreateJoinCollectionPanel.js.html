  7 /**
  8  *  This is the panel that allows a user to create or join a group on the settings
  9  *  page.
 10  *	@class
 11  *  @extends Panel
 12  **/
 13 var CreateJoinCollectionPanel = Panel.extend(
 14 	/**
 15 	 *	@scope	CreateJoinCollectionPanel.prototype
 16 	 **/
 17 {
 18     
 19     /**
 20     *  Initialize all stuffs.  We have an element for the input, as well as for the
 21     *  results.
 22     **/    
 23     initialize: function() {
 24         Panel.prototype.initialize.call(this);
 25         
 26         var params = this.options;
 27         
 28         var contents = this.contents;
 29         
 30         /**
 31          *  The input element where the user searches for a collection
 32          **/
 33         var inputElement = contents.find('.create_join_input');
 34         if(typeof(inputElement) == 'undefined' || inputElement.length == 0) {
 35             throw new Error('inputElement not found');
 36         }
 37         this.inputElement = inputElement;
 38         
 39         
 40         /**
 41          *  The Backbone.js collection of django Collection objects for search
 42          *  results.
 43          **/
 44         var searchResults = new CollectionSet;
 45         this.searchResults = searchResults;
 46         
 47         /* Another model object for the exact result (if one exists) */
 48         this.exactResult = null;
 49 
 50         /**
 51          *  The search result widget that displays the search results.
 52          **/
 53         var searchResultsWidget = new CollectionSearchResultsWidget({
 54             el: contents.find('.create_join_results'), 
 55             searchResults: searchResults, 
 56             panel: this, 
 57         });
 58         
 59         var createNewTemplate = $('#create_join_create_new_template');
 60         if(typeof(createNewTemplate) == 'undefined' || createNewTemplate.length == 0) {
 61             throw new Error('createNewTemplate not found');
 62         }
 63         this.createNewTemplate = createNewTemplate;
 64         
 65         var createResultsElement = $('#create_results');
 66         if(typeof(createResultsElement) == 'undefined' || 
 67             createResultsElement.length == 0) {
 68             throw new Error('createResultsElement not found');
 69         }
 70         this.createResultsElement = createResultsElement;
 71 
 72         /* This will hold a reference so we can keep track of the last Xhr */
 73         this.lastCreateJoinXhr = null;
 74 
 75         /* The current search term in the box */
 76         this.currentTerm = null;
 77         
 78         /* Initialize the auto complete behavior */
 79         this.initAutoCompleteBehavior();
 80         
 81         _.bindAll(this, "render");
 82         searchResults.bind('refresh', this.render);        
 83     },
 84     render: function() {
 85         this.createResultsElement.empty();
 86         
 87         /* There was no exact match, so we can create this collection */
 88         if(!this.exactResult && this.currentTerm != '') {
 89 
 90             /* "create new" button */
 91             var createNewElement = this.createNewTemplate.tmpl({
 92                 term: this.currentTerm, 
 93             });
 94             
 95             var createNewButton = new CreateNewCollectionButton({
 96                 container: createNewElement, 
 97                 newCollectionName: this.currentTerm,
 98                 panel: this,  
 99             });
100             
101             console.log(createNewElement.get(0));
102                         
103             /* Put in search results area */
104             this.createResultsElement.append(createNewElement);
105         }
106         
107         return this;
108     },
109     
110     events: {
111 
112     }, 
113     
114     /**
115      *  This should be called from init just once, to initialize the auto complete
116      *  behavior for the input element.
117      **/
118     initAutoCompleteBehavior: function() {
119         /* The actual jQuery UI autocomplete call */
120         this.inputElement.autocomplete({
121             minLength: 0,
122             source: function(me) {
123                 return function(request, response) {
124                     /* Our search term */
125                     var term = request.term;
126 
127                     /* If there is something to search for */
128                     if(term && term != '') {    
129                         /* Search for it */
130 
131                         //me.toggleLoadingNotification();
132                         me.lastCreateJoinXhr = $.getJSON(
133                             'search/'+term+'/', 
134                             {}, 
135                             function(data, status, xhr) {
136                                 /* If this was the last request made (ignore previous) */
137                                 if(xhr === me.lastCreateJoinXhr) {
138                                     me.currentTerm = term;
139                                     
140                                     me.exactResult = data.exact;
141                                     
142                                     me.searchResults.refresh(data.results);
143                                     
144                                    // me.toggleLoadingNotification();
145                                 }
146                             }
147                         );
148                     }
149                     /* Search is empty */
150                     else {
151                         me.currentTerm = term;
152                         me.searchResults.refresh([]);
153                     }
154                 };
155 
156             }(this),
157 
158             open: function(event, ui) {
159             },
160 
161             change: function(event, ui) {
162             },
163 
164             search: function(event, ui) {
165             }
166         });
167     },
168     /**
169      *  This should be called when the search form is to be reset.
170      **/
171     resetForm: function(){
172         /* Clear text field */
173         this.inputElement.val('');
174         this.currentTerm = '';
175         /* Clear auto complete */
176         this.searchResults.refresh([]);
177     }, 
178 
179 });
