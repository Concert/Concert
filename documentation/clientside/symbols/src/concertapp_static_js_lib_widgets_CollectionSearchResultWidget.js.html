  6 /**
  7  *  This widget is displayed for each collection in a collection search result.
  8  *  @class 
  9  *  @extend Widget
 10  **/
 11 var CollectionSearchResultWidget = Widget.extend(
 12 	/**
 13 	 *	@scope	CollectionSearchResultWidget.prototype
 14 	 **/
 15 {
 16     initialize: function() {
 17         Widget.prototype.initialize.call(this);
 18 
 19         var params = this.options;
 20         
 21         var page = this.panel.page;
 22         
 23         /* Save references to userRequests collection and user */
 24         this.userRequests = page.userRequests;        
 25         this.user = page.user;
 26                 
 27         _.bindAll(this, "render");
 28         this.render();
 29     },
 30     render: function() {
 31         Widget.prototype.render.call(this);
 32         
 33         return this;
 34     },
 35     events: {
 36         'click .collection_join_button': 'joinCollection', 
 37         'click .collection_info_button': 'showCollectionInfo', 
 38     },

 40      *  This will be executed when a user decides to join a collection.  Tell

 44         /* TODO: This is a bit confusing because the widget is calling the

 47     },
