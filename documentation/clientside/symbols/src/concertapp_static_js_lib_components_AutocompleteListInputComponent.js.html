  7 /**
  8  *  This is a component when we need to input a list of entities, with auto complete.
  9  *  @class
 10  *  @extends    Component
 11  **/
 12 var AutocompleteListInputComponent = Component.extend(
 13 /**
 14  *  @scope AudiocompleteListInputComponent.prototype
 15  **/
 16 {
 17     initialize: function() {
 18         Component.prototype.initialize.call(this);
 19 
 20         var params = this.options;
 21         
 22         var inputElement = params.inputElement;
 23         if(typeof(inputElement) == 'undefined') {
 24             throw new Error('params.inputElement is undefined');
 25         }
 26         this.inputElement = inputElement;
 27         
 28         _.bindAll(this, '_handle_input_keyup');
 29         inputElement.bind('keyup', this._handle_input_keyup);
 30         
 31         _.bindAll(this, '_handle_input_blur');
 32         inputElement.bind('blur', this._handle_input_blur);
 33 
 34         _.bindAll(this, "render");
 35     },
 36     
 37     /**
 38      *  Regexp used to split the input into separate tokens.
 39      **/
 40     TOKEN_SPLITTER: /\s*,\s*|\s{2,}/,
 41     SPACE_MATCH: /^\s*$/,
 42     
 43     /**
 44      *  Called when there is a keyup event on the input element.
 45      *
 46      *  @param  {Event}    e    The keyup event.
 47      **/
 48     _handle_input_keyup: function(e) {
 49         var inputElement = this.inputElement;
 50         
 51         var wordSplit = inputElement.val().split(AutocompleteListInputComponent.prototype.TOKEN_SPLITTER);
 52         
 53         /* If another token was created because a delimiter was entered */
 54         if(wordSplit.length > 1
 55             /* Or the return key was pressed */
 56             || e.keyCode == 13) {
 57             /* Grab previous token */
 58             var token = wordSplit[0];
 59             
 60             /* Remove everything from inputElement */
 61             this.inputElement.val('');
 62             
 63             /* If token was fo-real (not just another delimiter character) */
 64             if(token.length && !token.match(AutocompleteListInputComponent.prototype.SPACE_MATCH)) {
 65                 this._handle_new_token(token);
 66             }
 67             
 68         }
 69     },
 70     
 71     /**
 72      *  Called when there is a blur event on the input element.
 73      *
 74      *  @param  {Event}    e    The blur event.
 75      **/
 76     _handle_input_blur: function(e) {
 77         var inputElement = this.inputElement;
 78         
 79         var finalToken = inputElement.val().split(AutocompleteListInputComponent.prototype.TOKEN_SPLITTER);
 80         
 81         if(finalToken.length) {
 82             /* grab final token */
 83             var token = finalToken[0];
 84             
 85             /* remove everything from inputElement */
 86             this.inputElement.val('');
 87             
 88             if(token.length && !token.match(AutocompleteListInputComponent.prototype.SPACE_MATCH)) {
 89                 this._handle_new_token(token);
 90             }
 91         }
 92         
 93     }, 
 94     
 95     /**
 96      *  Called whenever input field is to be analyzed and new tokens are
 97      *  to be discovered.
 98      **/
 99     _look_for_tokens: function() {
100         
101     }, 
102     
103     /**
104      *  Handle a new token that was entered in the inputElement.
105      *
106      *  @param  {String}    token   The token that was entered.
107      **/
108     _handle_new_token: function(token) {
109         
110         return;
111     },
112 });
