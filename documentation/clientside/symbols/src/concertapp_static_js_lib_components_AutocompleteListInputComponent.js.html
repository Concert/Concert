  7 /**
  8  *  This is a component when we need to input a list of entities, with auto complete.
  9  *  @class
 10  *  @extends    Component
 11  **/
 12 var AutocompleteListInputComponent = Component.extend(
 13 /**
 14  *  @scope AudiocompleteListInputComponent.prototype
 15  **/
 16 {
 17     initialize: function() {
 18         Component.prototype.initialize.call(this);
 19 
 20         var params = this.options;
 21         
 22         var inputElement = params.inputElement;
 23         if(typeof(inputElement) == 'undefined') {
 24             throw new Error('params.inputElement is undefined');
 25         }
 26         this.inputElement = inputElement;
 27         
 28         _.bindAll(this, '_handle_input_keyup');
 29         inputElement.bind('keyup', this._handle_input_keyup);
 30 
 31         _.bindAll(this, "render");
 32     },
 33     
 34     /**
 35      *  Regexp used to split the input into separate tokens.
 36      **/
 37     TOKEN_SPLITTER: /[,\s]+/, 
 38 
 39     /**
 40      *  Called when there is a keyup event on the input element.
 41      *
 42      *  @param  {Event}    e    The keyup event.
 43      **/
 44     _handle_input_keyup: function(e) {
 45         var inputElement = this.inputElement;
 46         
 47         var wordSplit = inputElement.val().split(AutocompleteListInputComponent.prototype.TOKEN_SPLITTER);
 48         
 49         /* If another token was created */
 50         if(wordSplit.length > 1) {
 51             /* Grab previous token */
 52             var token = wordSplit[0];
 53             
 54             /* Remove everything from inputElement */
 55             inputElement.val('');
 56             
 57             /* If token was fo-real */
 58             if(token.length) {
 59                 this._handle_new_token(token);
 60             }
 61             
 62         }
 63     },
 64     
 65     /**
 66      *  Handle a new token that was entered in the inputElement.
 67      *
 68      *  @param  {String}    token   The token that was entered.
 69      **/
 70     _handle_new_token: function(token) {
 71         return;
 72     },
 73 });
