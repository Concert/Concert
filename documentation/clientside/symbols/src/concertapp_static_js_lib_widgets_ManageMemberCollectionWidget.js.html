  8 /**
  9  *  This class is for displaying the manage collection widget for a Concert
 10  *  Collection that the user is a member of.  If they are an administrator,
 11  *  this will be handled in the template.
 12  *  @class
 13  *  @extends    Widget
 14  **/
 15 var ManageMemberCollectionWidget = Widget.extend(
 16 	/**
 17 	 *	@scope	ManageMemberCollectionWidget.prototype
 18 	 **/
 19 {
 20     initialize: function() {
 21         Widget.prototype.initialize.call(this);
 22 
 23         var params = this.options;
 24         
 25         _.bindAll(this, "render");
 26         /* Render when users has changed */
 27         this.model.get('users').bind('add', this.render);
 28         this.model.get('users').bind('remove', this.render);
 29         this.render();
 30     },
 31     render: function() {
 32         Widget.prototype.render.call(this);
 33 
 34         return this;
 35     },
 36     events: {
 37         'click button.leave_collection': 'leave_collection_confirm'
 38     }, 
 39     /**
 40      *  When a user clicks the leave collection button
 41      **/
 42     leave_collection_confirm: function() {
 43         this.panel.page.leave_collection_with_confirm(this.model);
 44     }
 45 });
