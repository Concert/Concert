  7 /**
  8  *  A class for managing the user's set on the manage set page.  
  9  *  This is subclassed for the various types of set that the user can manage.
 10  *  @class
 11  *  @extends Panel
 12  **/
 13 var ManageTablePanel = Panel.extend(
 14 	/**
 15 	 *	@scope	ManageTablePanel.prototype
 16 	 **/
 17 {
 18     initialize: function() {
 19         Panel.prototype.initialize.call(this);
 20         
 21         var params = this.options;
 22         
 23         var contents = this.contents;
 24         
 25         var $ = jQuery;
 26         
 27         /**
 28          *  The set of objects we are managing on this table.
 29          **/
 30         var set = params.set;
 31         if(typeof(set) == 'undefined') {
 32             throw new Error('params.set is undefined');
 33         }
 34         this.set = set;
 35         
 36         /**
 37          *  The table element
 38          **/
 39         var table = $(contents).children('.table');
 40         if(typeof(table) == 'undefined') {
 41             throw new Error('table is undefined');
 42         }
 43         else if(table.length == 0) {
 44             throw new Error('table not found');
 45         }
 46         this.table = table;
 47         
 48         /**
 49          *  The table header (first row)
 50          **/
 51         var tableHeader = $(table).children('.table-header');
 52         if(typeof(tableHeader) == 'undefined') {
 53             throw new Error('tableHeader is undefined');
 54         }
 55         else if(tableHeader.length == 0) {
 56             throw new Error('tableHeader not found');
 57         }
 58         this.tableHeader = tableHeader;        
 59         
 60         
 61         _.bindAll(this, "render");
 62         set.bind('refresh', this.render);
 63         set.bind('add', this.render);
 64         set.bind('remove', this.render);
 65     },
 66     render: function() {
 67 
 68         /* Temporary fragment for all DOM additions */
 69         var frag = document.createDocumentFragment();
 70 
 71         /* The header will always be first in the table */
 72         frag.appendChild(this.tableHeader.get(0))
 73         
 74         this.set.each(function(panel, template, widgetClass, frag) {
 75             return function(obj){
 76                 /* Create a ManageCollectionWidget */
 77                 var widget = new widgetClass({
 78                     template: template, 
 79                     model: obj, 
 80                     panel: panel
 81                 });
 82                 frag.appendChild(widget.el);
 83             };
 84         }(this, this.widgetTemplate, this.widgetClass, frag));
 85         
 86         /* Update manage set list */
 87         this.table.empty().append(frag);
 88 
 89         return this;
 90     }
 91 });
