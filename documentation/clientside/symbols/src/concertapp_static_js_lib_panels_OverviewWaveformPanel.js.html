  7 /**
  8  *  This is the smaller waveform panel, which will be at the top underneath the 
  9  *  options bar on the "organization" page.
 10  *	@class
 11  *  @extends    WaveformPanel
 12  **/
 13 var OverviewWaveformPanel = WaveformPanel.extend(
 14 	/**
 15 	 *	@scope	OverviewWaveformPanel.prototype
 16 	 **/
 17 {
 18     initialize: function() {
 19         WaveformPanel.prototype.initialize.call(this)
 20         
 21         var playheadComponent = new OverviewWaveformPlayheadComponent({
 22             el: this.playheadElement,
 23             panel: this,
 24             audio: this.page.audio
 25         });
 26         /**
 27          *  Playhead for this waveform panel
 28          **/
 29         this.playheadComponent = playheadComponent;
 30         
 31         var highlighterContainerElement = $('#overview_waveform_panel_highlight_container');
 32         if(typeof(highlighterContainerElement) == 'undefined') {
 33             throw new Error('$(\'#overview_waveform_panel_highlight_container\') is undefined');
 34         }
 35         else if(highlighterContainerElement.length == 0) {
 36             throw new Error('highlighterContainerElement not found');
 37         }
 38         /**
 39          *  Highlighter container DOM element.
 40          **/
 41         this.highlighterContainerElement = highlighterContainerElement;
 42         
 43         /* Highlighter */
 44         var highlighter = new OverviewWaveformInteractionComponent({
 45             el: highlighterContainerElement, 
 46             panel: this 
 47         });
 48         /**
 49          *  Highlighter component for this panel
 50          **/
 51         this.highlighter = highlighter;
 52                 
 53         var segmentBarsContainerElement = $('#overview_waveform_panel_bottom');
 54         if(typeof(segmentBarsContainerElement) == 'undefined') {
 55             throw new Error('$(\'#overview_waveform_panel_bottom\') is undefined');
 56         }
 57         else if(segmentBarsContainerElement.length == 0) {
 58             throw new Error('segmentBarsContainerElement not found');
 59         }
 60         /**
 61          *  Container element for all of our segment bars.
 62          **/
 63         this.segmentBarsContainerElement = segmentBarsContainerElement;
 64         
 65         var segmentBarRowTemplate = $('#segment_bar_row_template');
 66         if(typeof(segmentBarRowTemplate) == 'undefined') {
 67             throw new Error('$(\'#segment_bar_row_template\') is undefined');
 68         }
 69         else if(segmentBarRowTemplate.length == 0) {
 70             throw new Error('segmentBarRowTemplate not found');
 71         }
 72         /**
 73          *  The template for each segment bar row
 74          **/
 75         this.segmentBarRowTemplate = segmentBarRowTemplate;
 76         
 77         var segmentBarTemplate = $('#segment_bar_template');
 78         if(typeof(segmentBarTemplate) == 'undefined') {
 79             throw new Error('$(\'#segment_bar_template\') is undefined');
 80         }
 81         else if(segmentBarTemplate.length == 0) {
 82             throw new Error('segmentBarTemplate not found');
 83         }
 84         /**
 85          *  The template for each segment bar.  left and width will change
 86          *  dynamically.
 87          **/
 88         this.segmentBarTemplate = segmentBarTemplate;
 89         
 90         
 91     }, 
 92     
 93     /**
 94      *  Called from page when an audio file is selected.
 95      *
 96      *  @param  {AudioFile}    selectedAudioFile    -   The audio file instance
 97      **/
 98     audio_file_selected: function(e, selectedAudioFile) {
 99         WaveformPanel.prototype.audio_file_selected.call(this, e, selectedAudioFile);
100         
101         this._load_waveform_image(
102             selectedAudioFile.get_waveform_src(10),
103             function(me, selectedAudioFile) {
104                 return function() {
105                     /* Tell highlighter */
106                     me.highlighter.audio_file_selected(selectedAudioFile);
107                     /* render segments */
108                     me.render_segment_bars(selectedAudioFile);
109                 }
110             }(this, selectedAudioFile)
111         );
112     
113         this.playheadComponent.update_speed();
114     }, 
115     
116     /**
117      *  Called from page when audio segment is selected.
118      *
119      *  @param  {AudioSegment}    selectedAudioSegment    - The audio segment
120      **/
121     audio_segment_selected: function(e, selectedAudioSegment) {
122         WaveformPanel.prototype.audio_segment_selected.call(this, e, selectedAudioSegment);
123         
124         var audioFile = selectedAudioSegment.get('audioFile');
125         
126         /* Load waveform image */
127         this._load_waveform_image(
128             audioFile.get_waveform_src(10),
129             /* Then */
130             function(me, selectedAudioSegment, audioFile) {
131                 return function() {
132                     /* Tell highlighter */
133                     me.highlighter.audio_segment_selected(selectedAudioSegment);
134                     /* render segments */
135                     me.render_segment_bars(audioFile)
136                 };
137             }(this, selectedAudioSegment, audioFile)
138         );
139 
140         this.playheadComponent.update_speed();
141 
142     }, 
143     
144     /**
145      *  Called from page when audio segment is deleted. 
146      *  Re-renders segment bars when a segment is deleted. 
147      *
148      *  @param  {AudioSegment}    deletedAudioSegment     - The audio segment
149      **/
150     audio_segment_deleted: function(deletedAudioSegment) {
151         var audioFile = deletedAudioSegment.get('audioFile');
152         this.render_segment_bars(audioFile);
153     },
154     
155     /**
156      *  The resolution of the waveform image (in pixels per second)
157      **/
158     get_resolution: function() {
159         /* Width of image is currently always 960 */
160         var width = 960;
161         
162         /* current duration of audio file */
163         var duration = this.audioFileDuration;
164         
165         return width/duration;
166     },
167     
168     /**
169      *  Set up all of the audio segment bars.
170      *
171      *  @param  {AudioFile}    audioFile    The file whose segments we are rendering
172      **/
173     render_segment_bars: function(audioFile) {
174         /* Segments for this audio file */
175         var segments = audioFile.get('segments');
176         
177         /* For each row, save SegmentBarWidgets */
178         var segmentBarWidgets = [];
179         
180         /* for each row, save the right most SegmentBarWidget */
181         var rightmostSegmentBarWidgets = [];
182         
183         /* Each row element */
184         var segmentRowElements = [];
185         
186         /* Templates we will need */
187         var segmentBarRowTemplate = this.segmentBarRowTemplate;
188         var segmentBarTemplate = this.segmentBarTemplate;
189         
190         /* DOM Elements */
191         var segmentBarsContainerElement = this.segmentBarsContainerElement;
192         /* Here we will store all of the new contents */
193         var segmentBarsContents = document.createDocumentFragment();
194         
195         /**
196          *  Determine if segment will fit on a given row. Called below.
197          **/
198         var segmentWillFitOnRow = function() {
199             /* get rightmost SegmentBarWidget */
200             var rightmost = rightmostSegmentBarWidgets[rowIndex];
201             
202             /* If this widget will fit after it */
203             if(widget.leftPx > rightmost.rightPx) {
204                 return true;
205             }
206             else {
207                 return false;
208             }
209         }
210         
211         /* for each segment (they are already sorted by startTime)*/
212         for(var s = 0, sl = segments.length; s < sl; s++) {
213             var seg = segments.at(s);
214             
215             /* Which row we are currently on */
216             var rowIndex = 0;
217             
218             /* Create widget for this segment */
219             var widget = new AudioSegmentBarWidget({
220                 panel: this, 
221                 template: segmentBarTemplate,
222                 model: seg 
223             }).render();
224             
225             while(true) {
226                 /* If this row has been created */
227                 if(segmentRowElements[rowIndex]) {
228                     
229                     /* If segment will fit on this row */
230                     if(segmentWillFitOnRow()) {
231                         /* put widget on this row */
232                         segmentRowElements[rowIndex].append(widget.el);
233                         
234                         /* it is now the rightmost */
235                         rightmostSegmentBarWidgets[rowIndex] = widget;
236                         
237                         /* We're done with this segment */
238                         break;
239                     }
240                     /* if it wont fit */
241                     else {
242                         /* go to next row */
243                         rowIndex++;
244                     }                
245                 }
246                 /* There are no segments in this row yet, we will fit */
247                 else {
248                     
249                     /* Create row */
250                     var rowElement = segmentBarRowTemplate.tmpl({
251                         row: rowIndex
252                     });
253                     
254                     /* Put widget in row */
255                     rowElement.append(widget.el);
256                     
257                     /* Save row */
258                     segmentRowElements[rowIndex] = rowElement;
259                     
260                     /* Put row in panel */
261                     segmentBarsContents.appendChild(rowElement.get(0));
262                     
263                     /* We're currently the only segment, therefore we are the 
264                     rightmost */
265                     rightmostSegmentBarWidgets[rowIndex] = widget;
266                     
267                     /* Done with this segment */
268                     break;
269                 }
270             }
271             
272         }
273         
274         /* replace old segment bars with new ones */
275         segmentBarsContainerElement.html(segmentBarsContents);
276     }, 
277 })
