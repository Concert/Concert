  7 /**
  8  *  This is the smaller waveform panel, which will be at the top underneath the 
  9  *  options bar on the "organization" page.
 10  *	@class
 11  *  @extends    WaveformPanel
 12  **/
 13 var OverviewWaveformPanel = WaveformPanel.extend(
 14 	/**
 15 	 *	@scope	OverviewWaveformPanel.prototype
 16 	 **/
 17 {
 18     initialize: function() {
 19         WaveformPanel.prototype.initialize.call(this)
 20         
 21         /* Instantiate widget for playhead */
 22         var playheadComponent = new OverviewWaveformPlayheadComponent({
 23             el: this.playheadContainerElement,
 24             panel: this,
 25             audio: this.page.audio
 26         });
 27         this.playheadComponent = playheadComponent;
 28         
 29         var highlighterContainerElement = $('#overview_waveform_panel_highlight_container');
 30         if(typeof(highlighterContainerElement) == 'undefined') {
 31             throw new Error('$(\'#overview_waveform_panel_highlight_container\') is undefined');
 32         }
 33         else if(highlighterContainerElement.length == 0) {
 34             throw new Error('highlighterContainerElement not found');
 35         }
 36         this.highlighterContainerElement = highlighterContainerElement;
 37         
 38         /* Highlighter */
 39         var highlighter = new OverviewWaveformHighlighterComponent({
 40             el: highlighterContainerElement, 
 41             panel: this 
 42         });
 43         this.highlighter = highlighter;
 44 
 45         $("#overview_waveform_panel_top").bind('click', function(me) {
 46             return function(e) {
 47                 me.handle_click(get_event_x(e));
 48             };
 49         }(this));
 50     }, 
 51     
 52     /**
 53      *  Called from page when an audio file is selected.
 54      *
 55      *  @param  {AudioFile}    selectedAudioFile    -   The audio file instance
 56      **/
 57     audio_file_selected: function(selectedAudioFile) {
 58         WaveformPanel.prototype.audio_file_selected.call(this, selectedAudioFile);
 59         
 60         this._load_waveform_image(
 61             selectedAudioFile.get_waveform_src(10),
 62             function(me, selectedAudioFile) {
 63                 return function() {
 64                     me.highlighter.audio_file_selected(selectedAudioFile);
 65                 }
 66             }(this, selectedAudioFile)
 67         );
 68     }, 
 69     
 70     /**
 71      *  Called from page when audio segment is selected.
 72      *
 73      *  @param  {AudioSegment}    selectedAudioSegment    - The audio segment
 74      **/
 75     audio_segment_selected: function(selectedAudioSegment) {
 76         WaveformPanel.prototype.audio_segment_selected.call(this, selectedAudioSegment);
 77         
 78         this.waveformImageElement.attr('src', selectedAudioSegment.get('audioFile').get_waveform_src(10));
 79     }, 
 80     
 81     /**
 82      *  The resolution of the waveform image (in pixels per second)
 83      **/
 84     get_resolution: function() {
 85         /* Width of image is currently always 898 */
 86         var width = 898;
 87         
 88         /* current duration of audio file */
 89         var duration = this.audioFileDuration;
 90         
 91         return width/duration;
 92     },
 93 
 94     handle_click: function(left) {
 95         //update audio's currentTime to location clicked
 96         var seconds = left/this.get_resolution();
 97         this.page.move_audio(seconds);
 98     }, 
 99     
100 })
