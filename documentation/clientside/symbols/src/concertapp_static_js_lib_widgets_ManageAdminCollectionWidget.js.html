  7 /**
  8  *  Manage a collection that the user is an administrator of
  9  *  @class
 10  *  @extends    Widget
 11  **/
 12 var ManageAdminCollectionWidget = Widget.extend(
 13 	/**
 14 	 *	@scope	ManageAdminCollectionWidget.prototype
 15 	 **/
 16 {
 17     initialize: function() {
 18         Widget.prototype.initialize.call(this);
 19 
 20         var params = this.options;
 21         
 22         /**
 23          *  The template for a collection request
 24          **/
 25         var collectionRequestWidgetTemplate = $('#collection_request_widget_template');
 26         if(typeof(collectionRequestWidgetTemplate) == 'undefined' || collectionRequestWidgetTemplate.length == 0) {
 27             throw new Error('collectionRequestWidgetTemplate not found');
 28         }
 29         this.collectionRequestWidgetTemplate = collectionRequestWidgetTemplate;
 30         
 31         
 32         /**
 33          *  The container for the collection requests
 34          **/
 35         this.requestContainer = null;
 36         
 37         
 38 
 39         _.bindAll(this, "render");
 40         /* Render when users has changed */
 41         this.model.get('users').bind('add', this.render);
 42         this.model.get('users').bind('remove', this.render);
 43         this.render();
 44     },
 45     render: function() {
 46         Widget.prototype.render.call(this);
 47         
 48         if(!this.requestContainer) {
 49             var requestContainer = $(this.el).children('.user_collection_requests');
 50             this.requestContainer = requestContainer;
 51         }
 52         else {
 53             var requestContainer = this.requestContainer;
 54         }
 55         
 56         var frag = document.createDocumentFragment();
 57         
 58         /*  We are now done rendering the collection information, but we need to 
 59             create a widget for every user who has requested to join the collection*/
 60         this.model.get('requests').each(function(parentWidget, template, frag){
 61             return function(req) {
 62                 var widget = new CollectionRequestWidget({
 63                     panel: parentWidget.panel, 
 64                     model: req, 
 65                     template: template
 66                 });
 67 
 68                 frag.appendChild(widget.el);
 69             };
 70         }(this, this.collectionRequestWidgetTemplate, frag));
 71         
 72         requestContainer.html(frag);
 73         
 74         return this;
 75     },
 76     events: {
 77         'click .delete_collection': 'delete_collection'
 78     },
 79     /**
 80      *  This is called when the user first clicks the delete button.  Let page
 81      *  handle it.
 82      **/
 83     delete_collection: function() {
 84         this.panel.page.deleteCollectionWithConfirm(this.model);
 85     }, 
 86 });
