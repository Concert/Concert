  7 
  8 var EditableModelTextComponent = Component.extend(
  9 	/**
 10 	 *	@scope	EditableModelTextComponent.prototype
 11 	 **/
 12 {
 13     initialize: function() {
 14         Component.prototype.initialize.call(this);
 15         
 16         var params = this.options;
 17         
 18         var el = $(this.el);
 19         
 20         /* The attribute we are changing on this model instance */
 21         var attr = params.attr;
 22         if(typeof(attr) == 'undefined') {
 23             throw new Error('params.attr is undefined');
 24         }
 25         this.attr = attr;
 26         
 27         /* Get input element (for changing the attribute) */
 28         var inputElement = el.children('input');
 29         if(typeof(inputElement) == 'undefined') {
 30             throw new Error('inputElement is undefined');
 31         }
 32         else if(inputElement.length == 0) {
 33             throw new Error('inputElement not found');
 34         }
 35         /* remove from DOM */
 36         this.inputElement = inputElement.detach();
 37         
 38         /* If there is more than one element left */
 39         if(el.children().length > 1) {
 40             throw new Error('There are too many elements in this.el');
 41         }
 42         
 43         /* Get other element */
 44         var displayElement = el.children().first();
 45         if(typeof(displayElement) == 'undefined') {
 46             throw new Error('el.children().first() is undefined');
 47         }
 48         this.displayElement = displayElement;
 49         
 50         /* Make sure proper classes are on elements */
 51         if(!displayElement.hasClass('editable_attribute')) {
 52             throw new Error('displayElement should have class "editable_attribute"');
 53         }
 54         
 55         if(!inputElement.hasClass('attribute_editor')) {
 56             throw new Error('inputElement should have class "attribute_editor"');
 57         }
 58         
 59         
 60         _.bindAll(this, 'attr_clicked');
 61         _.bindAll(this, 'attr_input_blurred');
 62     }, 
 63     
 64     events: {
 65         'blur .attribute_editor': 'attr_input_blurred',
 66         'click .editable_attribute': 'attr_clicked', 
 67     }, 
 68     
 69     /**
 70      *  Called from self when the attribute of the model is clicked.
 71      **/
 72     attr_clicked: function(){
 73         /* Switch to input box */
 74         $(this.el).html(this.inputElement);
 75 
 76         this.delegateEvents();
 77     }, 
 78     
 79     /**
 80      *  Called from self when new attribute field was blurred.  Might still be
 81      *  old name.  
 82      **/
 83     attr_input_blurred: function(e) {
 84 
 85         var inputElement = this.inputElement;        
 86         /* New attribute */
 87         var newAttr = inputElement.val();
 88         
 89         var model = this.model;
 90         var attr = this.attr;
 91         
 92         var oldAttr = model.get(attr);
 93         
 94         /* If name has changed */
 95         if(oldAttr != newAttr) {
 96             /* Save new name */
 97             model.save({
 98                 name: newAttr
 99             }, {
100                 /* If save fails */
101                 error_message: 'New '+attr+' was not saved.',
102                 error_callback: function(model, oldAttr) {
103                     /* Put back old attr */ 
104                     return function() {
105                         model.set({name: oldAttr});
106                     };
107                 }(model, oldAttr), 
108             });
109         }
110         
111         /* Change display element to new attr */
112         this.displayElement.html(newAttr);
113         
114         /* Put back display element */
115         $(this.el).html(this.displayElement);
116     },
117 });
