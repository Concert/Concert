  7 
  8 var EditableModelTextComponent = Component.extend(
  9 	/**
 10 	 *	@scope	EditableModelTextComponent.prototype
 11 	 **/
 12 {
 13     initialize: function() {
 14         Component.prototype.initialize.call(this);
 15         
 16         var params = this.options;
 17         
 18         var el = $(this.el);
 19         
 20         var attr = params.attr;
 21         if(typeof(attr) == 'undefined') {
 22             throw new Error('params.attr is undefined');
 23         }
 24         /**
 25          *  The name of the attribute we are changing on this model instance.
 26          **/
 27         this.attr = attr;
 28         
 29         var inputElement = el.children('input');
 30         if(typeof(inputElement) == 'undefined') {
 31             throw new Error('inputElement is undefined');
 32         }
 33         else if(inputElement.length == 0) {
 34             throw new Error('inputElement not found');
 35         }
 36         /**
 37          *  The input element that we will use to modify the attribute. (removed
 38          *  from DOM).
 39          **/
 40         this.inputElement = inputElement.detach();
 41         
 42         /* If there is more than one element left */
 43         if(el.children().length > 1) {
 44             throw new Error('There are too many elements in this.el');
 45         }
 46         
 47         /* Get other element */
 48         var displayElement = el.children().first();
 49         if(typeof(displayElement) == 'undefined') {
 50             throw new Error('el.children().first() is undefined');
 51         }
 52         /**
 53          *  The element we will use to display the value of this attribute. (removed
 54          *  from DOM).
 55          **/
 56         this.displayElement = displayElement;
 57         
 58         /* Make sure proper classes are on elements */
 59         if(!displayElement.hasClass('editable_attribute')) {
 60             throw new Error('displayElement should have class "editable_attribute"');
 61         }
 62         
 63         if(!inputElement.hasClass('attribute_editor')) {
 64             throw new Error('inputElement should have class "attribute_editor"');
 65         }
 66         
 67         /* Bind keyup events in input element to our keyup handler */
 68         _.bindAll(this, '_handle_input_keyup');
 69         inputElement.bind('keyup', this._handle_input_keyup);
 70         
 71         _.bindAll(this, 'attr_clicked');
 72         _.bindAll(this, 'attr_input_blurred');
 73     }, 
 74     
 75     events: {
 76         'blur .attribute_editor': 'attr_input_blurred',
 77         'click .editable_attribute': 'attr_clicked', 
 78     }, 
 79     
 80     /**
 81      *  Called from self when the attribute of the model is clicked.
 82      **/
 83     attr_clicked: function(){
 84         var inputElement = this.inputElement;
 85         
 86         /* Switch to input box */
 87         $(this.el).html(inputElement);
 88         
 89         /* Give input box focus */
 90         this.inputElement.focus();
 91 
 92         this.delegateEvents();
 93     }, 
 94     
 95     /**
 96      *  Called from inputElement keyup event handler.
 97      *
 98      *  @param  {Event}     e   The keyup event
 99      **/
100     _handle_input_keyup: function(e) {
101         
102         /* If this was a return key */
103         var keycode = e.keyCode;
104         if(keycode == 13) {
105 //            e.stopPropagation();
106             
107             /* Blur field */
108             this.inputElement.blur();
109             
110         }
111     }, 
112     
113     /**
114      *  Called from self when new attribute field was blurred.  Might still be
115      *  old name.  
116      **/
117     attr_input_blurred: function(e) {
118         var inputElement = this.inputElement;
119         var displayElement = this.displayElement;        
120         
121         /* New attribute */
122         var newAttr = inputElement.val();
123         
124         var model = this.model;
125         var attr = this.attr;
126         
127         var oldAttr = model.get(attr);
128         
129         /* If name has changed */
130         if(oldAttr != newAttr) {
131             /* Save new name */
132             model.save({
133                 name: newAttr
134             }, {
135                 /* If save fails */
136                 error_message: 'New '+attr+' was not saved.',
137                 error_callback: function(model, oldAttr) {
138                     /* Put back old attr */ 
139                     return function() {
140                         model.set({name: oldAttr});
141                     };
142                 }(model, oldAttr), 
143             });
144         }
145         
146         
147         /* Change display element to new attr text */
148         displayElement.html(newAttr);
149 
150         /* Put back display element */
151         $(this.el).html(displayElement);
152     },
153 });
