  9  *  deny buttons.
 10  *  @class
 11  *  @extends    Widget
 12  **/
 13 var CollectionRequestWidget = Widget.extend(
 14 	/**
 15 	 *	@scope	CollectionRequestWidget.prototype
 16 	 **/
 17 {
 18     initialize: function() {
 19         Widget.prototype.initialize.call(this);
 20 
 21         var params = this.options;
 22         
 23         
 24         /* Store a reference to the user widget */

 26 
 27         _.bindAll(this, "render");

 29     },

 45         'click .approve_request': 'approveRequestConfirm',
 46         'click .deny_request': 'denyRequestConfirm'

 49      *  This is called when the user clicks the deny request button.  Throw to
