  8 /**
  9  *  This class is for displaying the manage collection widget for a Concert
 10  *  Collection that the user is a member of.  If they are an administrator,
 11  *  this will be handled in the template.
 12  *  @class
 13  *  @extends    Widget
 14  **/
 15 var ManageCollectionWidget = Widget.extend(
 16 	/**
 17 	 *	@scope	ManageCollectionWidget.prototype
 18 	 **/
 19 {
 20     initialize: function() {
 21         Widget.prototype.initialize.call(this);
 22 
 23         var params = this.options;
 24         
 25         /**
 26          *  The template for a collection request
 27          **/
 28         var collectionRequestWidgetTemplate = $('#collection_request_widget_template');
 29         if(typeof(collectionRequestWidgetTemplate) == 'undefined' || collectionRequestWidgetTemplate.length == 0) {
 30             throw new Error('collectionRequestWidgetTemplate not found');
 31         }
 32         this.collectionRequestWidgetTemplate = collectionRequestWidgetTemplate;
 33         
 34         /**
 35          *  The container for the collection requests
 36          **/
 37         this.requestContainer = null;
 38         
 39         _.bindAll(this, "render");
 40         /* Render when users has changed */
 41         this.model.get('users').bind('add', this.render);
 42         this.model.get('users').bind('remove', this.render);
 43         this.render();
 44     },
 45     render: function() {
 46         Widget.prototype.render.call(this);
 47 
 48         return this;
 49     },
 50     events: {
 51         'click button.leave_collection': 'leave_collection_confirm'
 52     }, 
 53     /**
 54      *  When a user clicks the leave collection button
 55      **/
 56     leave_collection_confirm: function() {
 57         this.panel.page.leave_collection_with_confirm(this.model);
 58     }
 59 });
