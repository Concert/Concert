  7 var CommentsPanel = Panel.extend({
  8     initialize: function() {
  9         Panel.prototype.initialize.call(this);
 10         
 11         var params = this.options;
 12         
 13         var modelManager = params.modelManager;
 14         if(typeof(modelManager) == 'undefined') {
 15             throw new Error('params.modelManager is undefined');
 16         }
 17         /* Don't need to keep model manager reference around, just to bind events 
 18         this.modelManager = modelManager;*/
 19         
 20         /* The template for comment widgets */
 21         var commentTemplate = $('#comment_widget_template');
 22         if(typeof(commentTemplate) == 'undefined') {
 23             throw new Error('$(\'#comment_widget_template\') is undefined');
 24         }
 25         this.commentTemplate = commentTemplate;
 26         
 27         /* The template for the panel header */
 28         var headerTemplate = $('#comment_panel_header_template');
 29         if(typeof(headerTemplate) == 'undefined') {
 30             throw new Error('$(\'#comment_panel_header_template\') is undefined');
 31         }
 32         else if(headerTemplate.length == 0) {
 33             throw new Error('headerTemplate not found');
 34         }
 35         this.headerTemplate = headerTemplate;
 36         
 37     },
 38     
 39     /**
 40      *  When a segment is selected.
 41      *
 42      *  @param  {AudioSegment}    selectedAudioSegment    - The segment that was 
 43      *  selected.
 44      **/
 45     select_audio_segment: function(e, selectedAudioSegment) {
 46         
 47         /* display comments for this segment in contents */
 48         
 49         /* Fragment to hold all comments */
 50         var frag = document.createDocumentFragment();
 51         
 52         var template = this.commentTemplate;
 53         
 54         /* For each of this segment's comments */
 55         selectedAudioSegment.get('comments').each(function(frag, template, panel) {
 56             return function(obj) {
 57                 /* Create comment widget */
 58                 var widget = new CommentWidget({
 59                     model: obj,
 60                     template: template, 
 61                     panel: panel, 
 62                 });
 63                 
 64                 /* put in document fragment */
 65                 frag.appendChild(widget.render().el);
 66             };
 67         }(frag, template, this));
 68         
 69         /* Fill contents */
 70         this.contents.html(frag);
 71         
 72         /* Fill header */
 73         this.header.html(this.headerTemplate.tmpl(selectedAudioSegment.toJSON()));
 74     }, 
 75     
 76     /**
 77      *  When a file is selected.
 78      *
 79      *  @param  {AudioFile}    selectedAudioFile    -   The file that was selected.
 80      **/
 81     select_audio_file: function(e, selectedAudioFile) {
 82         /* just clear for now */
 83         this.contents.empty();
 84     }, 
 85     
 86 });
