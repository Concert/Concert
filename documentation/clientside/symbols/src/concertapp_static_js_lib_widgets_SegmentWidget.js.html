  7 /**
  8  *  This is the widget associated with an audio segment on the organize page.
  9  *	@class
 10  *  @extends    AudioListWidget
 11  **/
 12 var SegmentWidget = AudioListWidget.extend(
 13 	/**
 14 	 *	@scope	SegmentWidget.prototype
 15 	 **/
 16 {
 17     initialize: function() {
 18         AudioListWidget.prototype.initialize.call(this);
 19         
 20         var params = this.options;      
 21         
 22         
 23         this.render();
 24     }, 
 25     
 26     render: function() {
 27         AudioListWidget.prototype.render.call(this);
 28         
 29         var tagContainerElement = $(this.el).find('.bottom');
 30         var panel = this.panel;
 31         var frag = document.createDocumentFragment();
 32         
 33         /* For each tag */
 34         this.model.get('tags').each(function(tag) {
 35             /* Create tag widget */
 36             var widget = new TagWidget({
 37                 panel: panel, 
 38                 model: tag 
 39             });
 40             
 41             frag.appendChild(widget.render().el);
 42         });
 43         
 44         tagContainerElement.html(frag);
 45         
 46         return this;
 47     }, 
 48     /**
 49      *  When this widget's delete button is clicked.
 50      **/    
 51     _handle_delete_click: function() {
 52         AudioListWidget.prototype._handle_delete_click.call(this);
 53         
 54         this.panel.page.delete_audio_segment(this.model);
 55     }, 
 56 });
